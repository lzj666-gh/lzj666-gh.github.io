#include<bits/stdc++.h>
#define all(a) (a).begin(), (a).end()
#define file(a) { freopen(#a".in", "r", stdin); freopen(#a".out", "w", stdout); }
using namespace std;

vector<int>son[3][50]; // son[0]: i > j, son[1]: i = j, son[2]: i < j
struct Edge {
	int v, nxt;
}edge[1000];
int head[50], idx;
void add(int u, int v) {
	edge[++idx] = { v,head[u] };
	head[u] = idx;
}

vector<vector<int>>v;
int n; char opt;
bool sure[50];
int num[50];
bool vis[2][3][50];
int lim[2][50];

// 缩点相关变量
int belong[50]; // 每个点属于的等价类
vector<int> comp[50]; // 每个等价类包含的点
int comp_cnt; // 等价类数量
bool comp_vis[50]; // 用于DFS

// DFS找连通分量（相等关系的传递闭包）
void dfs(int u, int comp_id) {
    comp_vis[u] = true;
    belong[u] = comp_id;
    comp[comp_id].push_back(u);
    
    for (int v : son[1][u]) {
        if (!comp_vis[v]) {
            dfs(v, comp_id);
        }
    }
}

// 缩点：将所有相等的点合并为一个等价类
void compress() {
    memset(comp_vis, 0, sizeof(comp_vis));
    comp_cnt = 0;
    
    for (int i = 0; i < n; i++) {
        if (!comp_vis[i]) {
            dfs(i, comp_cnt);
            comp_cnt++;
        }
    }
    
    // 重新构建关系图（在等价类之间）
    vector<int> new_son[3][50];
    
    // 处理大于关系
    for (int i = 0; i < n; i++) {
        for (int j : son[0][i]) {
            int u = belong[i], v = belong[j];
            if (u != v) {
                new_son[0][u].push_back(v);
            }
        }
    }
    
    // 处理小于关系  
    for (int i = 0; i < n; i++) {
        for (int j : son[2][i]) {
            int u = belong[i], v = belong[j];
            if (u != v) {
                new_son[2][u].push_back(v);
            }
        }
    }
    
    // 去重
    for (int k = 0; k < 3; k++) {
        for (int i = 0; i < comp_cnt; i++) {
            sort(all(new_son[k][i]));
            new_son[k][i].erase(unique(all(new_son[k][i])), new_son[k][i].end());
        }
    }
    
    // 复制回原数组
    for (int i = 0; i < comp_cnt; i++) {
        for (int k = 0; k < 3; k++) {
            son[k][i] = new_son[k][i];
        }
    }
    
    n = comp_cnt; // 更新节点数为等价类数量
}

int32_t main() {
//	file(ex_b3);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n;
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < n; ++j) {
			cin >> opt;
			if (opt == '+')
				son[0][i].push_back(j),
				son[2][j].push_back(i);
			if (opt == '-')
				son[0][j].push_back(i),
				son[2][i].push_back(j);
			if (opt == '=')
				son[1][i].push_back(j),
				son[1][j].push_back(i);
		}
    
    // 进行缩点
    compress();
    
    // 记录原始a,b的位置
    int orig_a, orig_b;
    cin >> orig_a >> orig_b; 
    orig_a--; orig_b--;
    
    // 获取a,b在缩点后的等价类编号
    int a = belong[orig_a], b = belong[orig_b];

    // 原有的推理逻辑，现在在缩点后的图上进行
	for (int k = 0; k < n; ++k) {
		for (int i = 0; i < n; ++i) {
			for (auto& p : son[0][i])
				if (sure[p] && num[p] == 2)
					sure[i] = 1, num[i] = 3;
			for (auto& p : son[1][i])
				if (sure[p])
					sure[i] = 1, num[i] = num[p];
			for (auto& p : son[2][i])
				if (sure[p] && num[p] == 2)
					sure[i] = 1, num[i] = 1;
			if (son[0][i].size() && son[2][i].size())
				sure[i] = 1, num[i] = 2;
			for (auto& p : son[1][i])
				if (son[0][i].size() && son[2][p].size() || son[0][p].size() && son[2][i].size())
					sure[i] = 1, num[i] = 2;
		}
	}
    
	for (int i = 0; i < n; ++i)
		lim[0][i] = 1, lim[1][i] = 3;
	for (int i = 0; i < n; ++i)
		if (sure[i])
			lim[0][i] = lim[1][i] = num[i];
	for (int i = 0; i < n; ++i) {
		for (auto& p : son[0][i])
			lim[0][i] = max(lim[0][i], lim[0][p] + 1);
		for (auto& p : son[1][i])
			lim[0][i] = max(lim[0][i], lim[0][p]);
		for (auto& p : son[2][i])
			lim[0][i] = max(lim[0][i], lim[0][p] - 1);
		for (auto& p : son[0][i])
			lim[1][i] = min(lim[1][i], lim[1][p] + 1);
		for (auto& p : son[1][i])
			lim[1][i] = min(lim[1][i], lim[1][p]);
		for (auto& p : son[2][i])
			lim[1][i] = min(lim[1][i], lim[1][p] - 1);
	}

	// 计算vis数组
	for (auto& p : son[1][a])
		for (auto& q : son[2][p])
			vis[0][0][q] = 1;
	for (auto& q : son[2][a])
		vis[0][0][q] = 1;
	for (auto& p : son[1][a])
		vis[0][1][p] = 1;
	for (auto& p : son[1][a])
		for (auto& q : son[0][p])
			vis[0][2][q] = 1;
	for (auto& q : son[0][a])
		vis[0][2][q] = 1;
	if (sure[a])
		for (int i = 0; i < n; ++i)
			if (sure[i] && i != a)
				vis[0][int(num[i] <= num[a]) + (num[i] < num[a])][i] = 1;

	for (auto& p : son[1][b])
		for (auto& q : son[2][p])
			vis[1][0][q] = 1;
	for (auto& q : son[2][b])
		vis[1][0][q] = 1;
	for (auto& p : son[1][b])
		vis[1][1][p] = 1;
	for (auto& p : son[1][b])
		for (auto& q : son[0][p])
			vis[1][2][q] = 1;
	for (auto& q : son[0][b])
		vis[1][2][q] = 1;
	if (sure[b])
		for (int i = 0; i < n; ++i)
			if (sure[i] && i != b)
				vis[1][int(num[i] <= num[b]) + (num[i] < num[b])][i] = 1;

	// 统计答案，注意现在统计的是等价类之间的比较
	int ans[3] = { 0 };
	for (int i = 0; i < n; ++i)
		for (int j = i + 1; j < n; ++j)
			if (i != a && i != b && j != a && j != b) {
                // 计算这对等价类代表的实际点对数量
                int cnt_i = comp[i].size();
                int cnt_j = comp[j].size();
                int pair_count = cnt_i * cnt_j;
                
				if (vis[0][0][i] && vis[1][0][j] || vis[0][0][j] && vis[1][0][i])
					ans[0] += pair_count;
				else if (vis[0][0][i] && vis[1][1][j] || vis[0][0][j] && vis[1][1][i])
					ans[0] += pair_count;
				else if (vis[0][1][i] && vis[1][0][j] || vis[0][1][j] && vis[1][0][i])
					ans[0] += pair_count;
				else if (lim[0][i] + lim[0][j] > lim[1][a] + lim[1][b])
					ans[0] += pair_count;
				else if (vis[0][1][i] && vis[1][1][j] || vis[0][1][j] && vis[1][1][i])
					ans[1] += pair_count;
				else if (vis[0][1][i] && vis[1][2][j] || vis[0][1][j] && vis[1][2][i])
					ans[2] += pair_count;
				else if (vis[0][2][i] && vis[1][1][j] || vis[0][2][j] && vis[1][1][i])
					ans[2] += pair_count;
				else if (vis[0][2][i] && vis[1][2][j] || vis[0][2][j] && vis[1][2][i])
					ans[2] += pair_count;
				else if (lim[1][i] + lim[1][j] < lim[0][a] + lim[0][b])
					ans[2] += pair_count;
            }

	cout << ans[2] << endl;
    
	if (sure[a] && sure[b]) {
		int cnt[4] = { 0 }, cntt[8] = { 0 };
		for (int i = 0; i < n; ++i)
			if (sure[i]) {
                // 累加等价类中点的数量
                cnt[num[i]] += comp[i].size();
            }
                
		cnt[num[a]] -= comp[a].size(); 
        cnt[num[b]] -= comp[b].size();
        
		for (int i = 1; i <= 3; ++i)
			cntt[i + i] += cnt[i] * (cnt[i] - 1) / 2;
		for (int i = 1; i <= 3; ++i)
			for (int j = i + 1; j <= 3; ++j)
				cntt[i + j] += cnt[i] * cnt[j];
		cout << cntt[num[a] + num[b]] << endl;
	}
	else {
        // 对于不确定的情况，需要考虑等价类的大小
        int total = 0;
        for (int i = 0; i < n; ++i)
            for (int j = i + 1; j < n; ++j)
                if (i != a && i != b && j != a && j != b && 
                   (vis[0][1][i] && vis[1][1][j] || vis[0][1][j] && vis[1][1][i]))
                    total += comp[i].size() * comp[j].size();
        cout << total << endl;
    }
    
	cout << ans[0] << endl;
	return 0;
}
