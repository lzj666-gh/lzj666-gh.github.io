#include<bits/stdc++.h>
#define int long long
using namespace std;
int n, m, s, w[200000 + 5], v[200000 + 5], l[200000 + 5], r[200000 + 5], t[200000 + 5];
int sum[2][200000 + 5];
int check(int W) {
	for (int i = 1; i <= n; ++i)
		sum[0][i] = sum[0][i - 1] + (w[i] >= W),
		sum[1][i] = sum[1][i - 1] + (w[i] >= W) * v[i];
	int y = 0;
	for (int i = 1; i <= m; ++i)
		y += (sum[0][r[i]] - sum[0][l[i] - 1]) * (sum[1][r[i]] - sum[1][l[i] - 1]);
	return y - s;
}
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m >> s;
	for (int i = 1; i <= n; ++i) cin >> w[i] >> v[i];
	for (int i = 1; i <= m; ++i)
		cin >> l[i] >> r[i];
	int l = -1, r = 1e6 + 5;
	while (l < r - 1) {
		int mid = l + r >> 1;
		if (check(mid) > 0) l = mid;
		else r = mid;
	}
	cout << min({ abs(check(l)),abs(check(r)),abs(check(l - 1)),abs(check(r + 1l)) }) << endl;
	return 0;
}
