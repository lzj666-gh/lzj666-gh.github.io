#include<bits/stdc++.h>
#ifdef _DEBUG
#define __builtin_popcount _Popcount
#endif
#define int long long
#define all(a) (a).begin(), (a).end()
#define file(a) { freopen(#a".in", "r", stdin); freopen(#a".out", "w", stdout); }
using namespace std;
int n, m, q, a[2000000 + 5], anss[100], ans;
int32_t main() {
	// file(b);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n;
	for (int i = 1; i <= n; ++i)
		cin >> a[i];
	while ((1ll << m) <= n) ++m;
	int S = 0;
	for (int i = 1; i <= n; ++i) S += a[i];
	for (int i = 1; i <= n; ++i)
		for (int b = 0; b < m; ++b)
			if ((i >> b) & 1) anss[b] += a[i];
	for (int b = 0; b < m; ++b)
		ans += 2 * anss[b] * (S - anss[b]);
	cout << ans << endl; cin >> q;
	while (q--) {
		int x, y; cin >> x >> y;
		int gap = y - a[x];
		a[x] = y; S += gap;
		for (int b = 0; b < m; ++b)
			if ((x >> b) & 1) anss[b] += gap;
		ans = 0;
		for (int b = 0; b < m; ++b)
			ans += 2 * anss[b] * (S - anss[b]);
		cout << ans << endl;
	}
	return 0;
}
