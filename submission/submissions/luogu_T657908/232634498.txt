#include<bits/stdc++.h>
using namespace std;
int32_t main() {
    cin.tie(0)->sync_with_stdio(false);
    int T; cin >> T;
    while (T--) {
        static int n, c[100 + 5];
        cin >> n;
        for (int i = 1; i <= n; ++i) cin >> c[i];
        long long ans = 0;
        ans += c[1];
        for (int i = 2; i <= n; ++i) {
            ans += 2;
            switch (i % 10) {
            case 1: {
                ans += ceil((c[i] - 6) / 10.0);
                break;
            }
            case 2: {
                ans += ceil((c[i] - 3) / 10.0);
                break;
            }
            case 3: {
                ans += ceil((c[i] - 2) / 10.0);
                break;
            }
            case 6: {
                ans += ceil((c[i] - 2) / 10.0);
                break;
            }
            }
        }
        cout << ans << endl;
    }
    return 0;
}
