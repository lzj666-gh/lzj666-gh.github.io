#include<bits/stdc++.h>
#define int long long
using namespace std;
int n, t[10000000 + 5];
queue<int>q1, q2;
int get(queue<int>& q) {
	int k = q.front();
	q.pop();
	return k;
}
int get() {
	if (q1.empty()) return get(q2);
	if (q2.empty()) return get(q1);
	if (q1.front() < q2.front()) return get(q1);
	return get(q2);
}
signed main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n;
	for (int i = 1, a; i <= n; ++i) cin >> a, t[a]++;
	for (int i = 1; i <= 1e5; ++i)
		while (t[i]--)
			q1.push(i);
	int ans = 0;
	for (int i = 1; i < n; ++i) {
		int x = get(), y = get();
		ans += x += y; q2.push(x);
	}
	cout << ans << endl;
	return 0;
}
