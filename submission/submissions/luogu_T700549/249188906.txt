constexpr int mod = 1e9 + 7;
int clac(vector<int>v, int pos) {
	if (pos == -1) return 0;
	if (v.size() <= 1) return 0;
	bool flg1 = 1, flg2 = 1; int pos1 = 0, pos2 = 0;
	for (int i = 1; i < v.size(); ++i)
		if (!(v[i - 1] & (1 << pos)) && (v[i] & (1 << pos))) pos1 = i, flg2 = 0;
		else if ((v[i - 1] & (1 << pos)) && !(v[i] & (1 << pos))) pos2 = i, flg1 = 0;
	if (flg1) return clac(vector<int>(v.begin(), v.begin() + pos1), pos - 1) | clac(vector<int>(v.begin() + pos1, v.end()), pos - 1);
	if (flg2) return clac(vector<int>(v.begin(), v.begin() + pos2), pos - 1) | clac(vector<int>(v.begin() + pos2, v.end()), pos - 1) | 1 << pos;
	return -1;
}
bool check(vector<int>v, int res) {
	for (int i = 1; i < v.size(); ++i)
		if ((v[i - 1] ^ res) > (v[i] ^ res))
			return false;
	return true;
}
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	int n; cin >> n; vector<int>v(n);
	for (int i = 0; i < n; ++i) cin >> v[i];
	int q; cin >> q; cout << clac(v, 30) << endl;
	while (q--) {
		int x, k; cin >> x >> k; v[x - 1] = k; int ans = clac(v, 30);
		assert(ans == -1 || check(v, ans));
		cout << ans << endl;
	}
	return 0;
}
