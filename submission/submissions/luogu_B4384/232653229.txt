#include<bits/stdc++.h>
using namespace std;
int32_t main() {
    cin.tie(0)->sync_with_stdio(false);
    int T; cin >> T;
    while (T--) {
        static int n, c[100 + 5];
        cin >> n;
        for (int i = 1; i <= n; ++i) cin >> c[i];
        long long ans = 0;
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= c[i]; ++j) {
                if (i == 1 || j == 1 || j == c[i] || i * j % 10 == 6) {
                    ans++;
                }
            }
        }
        cout << ans << endl;
    }
    return 0;
}
