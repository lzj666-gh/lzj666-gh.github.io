#include<bits/stdc++.h>
using namespace std;
int main() {
	cin.tie(0)->sync_with_stdio(false);
	int T; cin >> T;
	while (T--) {
		int a, b, c, cnt = 0; cin >> a >> b >> c;
		for (int i = 1; i <= a; ++i) {
			if (a % i == 0) {
				if ((c - a / i) % b == 0) {
					cnt++;
				}
			}
		}
		cout << cnt << endl;
	}
	return 0;
}
