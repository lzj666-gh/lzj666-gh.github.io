#include<bits/stdc++.h>
using namespace std;
int main() {
	cin.tie(0)->sync_with_stdio(false);
	int sa, sb, sc; cin >> sa >> sb >> sc;
	if (sa >= 30 && sa <= 70) {
		cout << sa << endl;
		return 0;
	}
	if (sb >= 30 && sb <= 70) {
		if (sc <= 70 && sc >= 30) {
			cout << min(sb, sc) << endl;
			return 0;
		}
		cout << sb << endl;
		return 0;
	}
	if (sc <= 70 && sc >= 30) {
		cout << sc << endl;
		return 0;
	}
	cout << 0 << endl;
	return 0;
}
