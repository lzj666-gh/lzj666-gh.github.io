#include<bits/stdc++.h>

using namespace std;
constexpr int mod = 1e9 + 7;
int n, m, s[200000 + 5], f[200000 + 5];
bool check(int k) {
	for (int i = 1, j = 1, r = -1; i <= n; ++i) {
		if (s[i] > r) {
			if (j > m) return false;
			while (j <= m + 1) {
				if (j == m + 1) return false;
				if (s[i] < f[j]) {
					if (f[j] - k <= s[i] && s[i] <= f[j]) {
						r = f[j]; j++;
						break;
					}
				}
				else {
					if (f[j] <= s[i] && s[i] <= f[j] + k) {
						r = f[j] + k; j++;
						break;
					}
				}
				j++;
			}
		}
		//cerr << "checking " << i << " : ok. with farthest " << r << "(next machine " << j << ")" << endl;
	}
	return true;
}
void output(int k) {
	for (int i = 1, j = 1, r = -1; i <= n; ++i) {
		if (s[i] > r) {
			while (j <= m + 1) {
				if (s[i] < f[j]) {
					if (f[j] - k <= s[i] && s[i] <= f[j]) {
						r = f[j]; j++;
						cout << 'L';
						break;
					}
				}
				else {
					if (f[j] <= s[i] && s[i] <= f[j] + k) {
						r = f[j] + k; j++;
						cout << 'R';
						break;
					}
				}
				j++;
				cout << 'R';
			}
		}
		if (i == n) while (j++ <= m) cout << 'R' << endl;
	}
}
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m; f[m + 1] = INT_MAX;
	for (int i = 1; i <= m; ++i) cin >> f[i];
	for (int i = 1; i <= n; ++i) cin >> s[i];
	int l = -1, r = s[n], mid;
	//for (int i = l; i <= r; ++i) cout << check(i) << ' '; cout << endl;
	if (!check(r)) {
		cout << -1 << endl;
		return 0;
	}
	while (l < r - 1)
		if (check(mid = l + r >> 1)) r = mid;
		else l = mid;
	cout << r << endl;
	output(r);
	return 0;
}
