constexpr int mod = 19990721;
struct Matix {
	int num[2][2];
	Matix() : num() {}
	int* operator[](int p) {
		return num[p];
	}
	const int* operator[](int p)const {
		return num[p];
	}
	friend Matix operator*(const Matix& a, const Matix& b) {
		static Matix c; c.num[0][0] = c.num[0][1] = c.num[1][0] = c.num[1][1] = 0;
		for (int i = 0; i < 2; ++i)
			for (int j = 0; j < 2; ++j)
				for (int k = 0; k < 2; ++k)
					c[i][j] = (c[i][j] + 1ll * a[i][k] * b[k][j]) % mod;
		return c;
	}
};
struct Seg_Tree {
#define mid ((l + r) >> 1)
#define lson tree[p].ls, l, mid
#define rson tree[p].rs, mid + 1, r
#define self p, l, r
	struct Node {
		Matix res;
		int ls, rs; bool flg;
		Node() : res(), ls(), rs(), flg() {}
	}tree[4000000 << 2 | 3];
	int idx = 1;
	void measure(int& p, long long l, long long r) {
		if (p) return;
		p = ++idx;
		tree[p].res[0][0] = 1;
		tree[p].res[0][1] = (r - l + 1) % mod;
		tree[p].res[1][0] = 0;
		tree[p].res[1][1] = 1;
	}
	void pushup(int p, long long l, long long r) {
		measure(lson); measure(rson);
		tree[p].res = tree[tree[p].ls].res * tree[tree[p].rs].res;
	}
	void pushdown(int p, long long l, long long r) {
		if (tree[p].flg) {
			tree[p].flg = 0; measure(lson); measure(rson);
			swap(tree[tree[p].ls].res[0][0], tree[tree[p].ls].res[1][1]);
			swap(tree[tree[p].ls].res[0][1], tree[tree[p].ls].res[1][0]);
			swap(tree[tree[p].rs].res[0][0], tree[tree[p].rs].res[1][1]);
			swap(tree[tree[p].rs].res[0][1], tree[tree[p].rs].res[1][0]);
			tree[tree[p].ls].flg ^= 1; tree[tree[p].rs].flg ^= 1;
		}
	}
	void reverse(int p, long long l, long long r, long long x, long long y) {
		if (x <= l && r <= y) {
			tree[p].flg ^= 1;
			swap(tree[p].res[0][0], tree[p].res[1][1]);
			swap(tree[p].res[0][1], tree[p].res[1][0]);
			return;
		}
		pushdown(self); measure(lson); measure(rson);
		if (x <= mid) reverse(lson, x, y);
		if (mid < y) reverse(rson, x, y);
		pushup(self);
	}
	Matix query(int p, long long l, long long r, long long x, long long y) {
		if (x <= l && r <= y) return tree[p].res;
		pushdown(self); measure(lson); measure(rson);
		if (x <= mid && mid < y) return query(lson, x, y) * query(rson, x, y);
		if (x <= mid) return query(lson, x, y);
		if (mid < y) return query(rson, x, y);
	}
