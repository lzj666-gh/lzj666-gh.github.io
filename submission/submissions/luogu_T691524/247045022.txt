constexpr int mod = 19990721;
struct Matix {
	int num[2][2];
	int* operator[](int p) { return num[p]; }
	const int* operator[](int p) const { return num[p]; }
	Matix(int len = 1) { num[0][0] = 1; num[0][1] = len; num[1][0] = 0; num[1][1] = 1; }
	Matix(bool, int len) { num[0][0] = 1; num[0][1] = 0; num[1][0] = len; num[1][1] = 1; }
	void operator=(const Matix&oth) { memcpy(num, oth.num, sizeof num); }
	strong_ordering operator<=>(const Matix& oth)const = default;
	friend Matix operator*(const Matix& a, const Matix& b) {
		static Matix c; c.num[0][0] = c.num[0][1] = c.num[1][0] = c.num[1][1] = 0;
		for (int i = 0; i < 2; ++i)
			for (int j = 0; j < 2; ++j)
				for (int k = 0; k < 2; ++k)
					c[i][j] = (c[i][j] + 1ll * a[i][k] * b[k][j]) % mod;
		return c;
	}
};
struct Seg_Tree {
#define mid (l + r >> 1)
#define lson tree[p].ls, l, mid
#define rson tree[p].rs, mid + 1, r
#define self p, l, r
	struct Node {
		Matix res; int ls, rs; bool flg;
		Node() :res(), ls(), rs(), flg() {}
	}tree[14200000];
	int idx = 1;
	void measure(int& p, long long l, long long r) {
		if (p) return;
		p = ++idx;
		tree[p].res = Matix((r - l + 1) % mod);
	}
	void pushup(int p, long long l, long long r) {
		if (!tree[p].ls && !tree[p].rs) {
			tree[p].res = Matix((r - l + 1) % mod);
			return;
		}
		if (!tree[p].rs) tree[0].res = Matix((r - mid) % mod);
		if (!tree[p].ls) tree[0].res = Matix((mid - l + 1) % mod);
		tree[p].res = tree[tree[p].ls].res * tree[tree[p].rs].res;
	}
	void swap(int p) {
		tree[p].res[0][0] ^= tree[p].res[1][1] ^= tree[p].res[0][0] ^= tree[p].res[1][1];
		tree[p].res[0][1] ^= tree[p].res[1][0] ^= tree[p].res[0][1] ^= tree[p].res[1][0];
	}
	void pushdown(int p, long long l, long long r) {
		if (tree[p].flg) {
			tree[p].flg = 0; measure(lson); measure(rson);
			swap(tree[p].ls); swap(tree[p].rs);
			tree[tree[p].ls].flg ^= 1; tree[tree[p].rs].flg ^= 1;
		}
	}
	void reverse(int p, long long l, long long r, long long x, long long y) {
		if (x <= l && r <= y) { tree[p].flg ^= 1; swap(p); return; }
		pushdown(self);
		if (x <= mid) measure(lson), reverse(lson, x, y);
		if (mid < y)  measure(rson), reverse(rson, x, y);
		pushup(self);
	}
	Matix query(int p, long long l, long long r, long long x, long long y) {
		if (x <= l && r <= y) return tree[p].res;
		if (tree[p].res == Matix((r - l + 1) % mod))
			return Matix((min(r, y) - max(l, x) + 1) % mod);
		if (tree[p].res == Matix(false, (r - l + 1) % mod))
			return Matix(false, (min(r, y) - max(l, x) + 1) % mod);
		pushdown(self);
		if (x <= mid) measure(lson); if (mid < y) measure(rson);
		if (x <= mid && mid < y) return query(lson, x, y) * query(rson, x, y);
		if (x <= mid) return query(lson, x, y);
		if (mid < y) return query(rson, x, y);
		return Matix(true, 0);
	}
}t;
