#include<bits/stdc++.h>
#define int long long
#define all(a) (a).begin(), (a).end()
#define file(a) { freopen(#a".in", "r", stdin); freopen(#a".out", "w", stdout); }
using namespace std;
int n, q;
struct Edge {
	int u, v, w, nxt;
	friend bool operator<(const Edge& a, const Edge& b) {
		return a.w > b.w;
	}
}edge[100000 + 5];
int head[100000 + 5];
void add(int i) {
	edge[i].nxt = head[edge[i].u];
	head[edge[i].u] = i;
}
int siz[100000 + 5], fa[100000 + 5];
struct Query {
	int k, v, idx;
	friend bool operator<(const Query& a, const Query& b) {
		return a.k > b.k;
	}
}qry[100000 + 5];
int find(int x) { return fa[x] == x ? x : fa[x] = find(fa[x]); }
int ans[100000 + 5];
int32_t main() {
//	file(lockdown);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> q;
	for (int i = 1, u, v, w; i < n; ++i)
		cin >> u >> v >> w,
		edge[i].u = u, edge[i].v = v, edge[i].w = w;
	sort(edge + 1, edge + n);
	iota(fa + 1, fa + n + 1, 1);
	fill(siz + 1, siz + n + 1, 1);
	for (int i = 1; i <= q; ++i) cin >> qry[i].k >> qry[i].v, qry[i].idx = i;
	sort(qry + 1, qry + q + 1);
	for (int i = 1, j = 1; i <= q; ++i) {
		while (edge[j].w >= qry[i].k && j) {
			int u = find(edge[j].u), v = find(edge[j].v);
			siz[u] += siz[v];
			fa[v] = u;
			j++;
		}
		ans[qry[i].idx] = siz[find(qry[i].v)] - 1;
#ifdef _DEBUG
		cerr << "in query " << qry[i].k << endl;
		for (int i = 1; i <= n; ++i) cout << fa[i] << " \n"[i == n];
		for (int i = 1; i <= n; ++i) cout << (fa[i] == i ? siz[i] : 0) << " \n"[i == n];
#endif
	}
	for (int i = 1; i <= q; ++i) cout << ans[i] << endl;
	return 0;
}
