#include<bits/stdc++.h>
#define int long long
using namespace std;
struct Point {
	int x, y, idx;
}p[8000 + 5];
int n;
vector<vector<int>> pos;
int check(const Point& x, const Point& y) {
	return (x.y - p[1].y) * (y.x - p[1].x) - (y.y - p[1].y) * (x.x - p[1].x);
}
signed main(signed argc, char** argv) {
//	if (argc >= 2) freopen(argv[1], "r", stdin);
//	else freopen("a.in", "r", stdin);
//	if (argc >= 3) freopen(argv[2], "w", stdout);
//	else freopen("a.out", "w", stdout);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n; n++; 
	for (int i = 1; i <= n; ++i) cin >> p[i].x >> p[i].y, p[i].idx = i;
	int idx = 1;
	for (int i = 1; i <= n; ++i)
		if (p[idx].x > p[i].x || (p[idx].x == p[i].x && p[i].y < p[idx].y))
			idx = i;
	swap(p[1], p[idx]);
	sort(p + 2, p + n + 1, [](const Point& x, const Point& y)->bool { int res = check(x, y); return res == 0 ? x.x < y.x : res < 0; });
	pos.emplace_back();
	pos.back().push_back(2);
	for (int i = 3; i <= n; ++i)
		if (check(p[i], p[i - 1]) == 0)
			pos.back().push_back(i);
		else
			pos.emplace_back(),
			pos.back().push_back(i);
	if (pos.size() == 1)
		cout << -1 << endl;
	else {
		reverse(pos.back().begin(), pos.back().end());
		cout << p[1].idx << ' ';
		for (auto& k : pos)
			for (auto& q : k)
				cout << p[q].idx << ' ';
	}
	return 0;
}
