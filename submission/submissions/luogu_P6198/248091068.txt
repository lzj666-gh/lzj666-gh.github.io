#include<bits/stdc++.h>
#define file(a) freopen(#a".in", "r", stdin), freopen(#a".out", "w", stdout)
using namespace std;
int n;
struct FWT {
	int s[1000000 + 5];
	void add(int x, int k) {
		while (x <= n) {
			s[x] += k;
			x += x & ~x + 1;
		}
	}
	void add(int l, int r, int k) {
		add(l, k); add(r + 1, -k);
	}
	int query(int x) {
		int res = 0;
		while (x) {
			res += s[x];
			x -= x & ~x + 1;
		}
		return res;
	}
};
struct Seg_Tree {
#define lc(p) (p << 1)
#define rc(p) (p << 1 | 1)
#define mid (l + r >> 1)
#define lson lc(p), l, mid
#define rson rc(p), mid + 1, r
#define self p, l, r
	struct Node {
		int sum, mn, tag;
		Node() :sum(), mn(), tag() {}
	}tree[1000000 << 2 | 3];
	void pushup(int p) {
		tree[p].sum = tree[lc(p)].sum + tree[rc(p)].sum;
		tree[p].mn = min(tree[lc(p)].mn, tree[rc(p)].mn);
	}
	void add(int p, int l, int r, int k) {
		tree[p].sum += (r - l + 1) * k;
		tree[p].tag += k; tree[p].mn += k;
	}
	void pushdown(int p, int l, int r) {
		if (tree[p].tag) {
			add(lson, tree[p].tag);
			add(rson, tree[p].tag);
			tree[p].tag = 0;
		}
	}
	void add(int p, int l, int r, int x, int y, int k) {
		if (x <= l && r <= y) return add(self, k);
		if (r < x || y < l) return;
		pushdown(self);
		add(lson, x, y, k);
		add(rson, x, y, k);
		pushup(p);
	}
	int querysum(int p, int l, int r, int x, int y) {
		if (x <= l && r <= y) return tree[p].sum;
		if (r < x || y < l) return 0;
		pushdown(self);
		return querysum(lson, x, y) + querysum(rson, x, y);
	}
	int querymin(int p, int l, int r, int x, int y) {
		if (x <= l && r <= y) return tree[p].mn;
		if (r < x || y < l) return 1e9;
		pushdown(self);
		return min(querymin(lson, x, y), querymin(rson, x, y));
	}
}t;
int fst[1000000 + 5], lst, idx;
int main() {
	//file(stack);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n;
	for (int i = 1; i <= n; ++i) {
		int x; cin >> x;
		if (x == -1) {
			x = lst + 1;
		}
		if (x > lst) {
			t.add(1, 1, n, i, i, ++idx);
			lst = x; fst[x] = i;
			continue;
		}
		int num = t.querymin(1, 1, n, fst[x], i - 1);
		t.add(1, 1, n, fst[x], i - 1, 1);
		t.add(1, 1, n, i, i, num); ++idx;
		lst = x;
	}
	for (int i = 1; i <= n; ++i) cout << t.querysum(1, 1, n, i, i) << " \n"[i == n];
	return 0;
}
