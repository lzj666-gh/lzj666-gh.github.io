#include<bits/stdc++.h>
using namespace std;
constexpr int mod = 1e7 + 9;
string s;
int main() {
    cin >> s;
    long long pre = 0, tot = count(s.begin(), s.end(), 'c');
    long long ans = 0;
    for (char& c : s) {
        if (c == 'c') pre++;
        if (c == 'y') ans += pre * (tot - pre) % mod, ans %= mod;
    }
    cout << ans << endl;
    return 0;
}
