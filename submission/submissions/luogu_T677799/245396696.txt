#include<bits/stdc++.h>
using namespace std;
int main() {
	cin.tie(0)->sync_with_stdio(false);
	long long n, m, k; int cnt[4] = { 0 };
	cin >> n >> m >> k;
	for (int i = 1, x, y; i <= k; ++i) {
		cin >> x >> y;
		if ((x & 1) && (y & 1)) cnt[0]++;
		if (!(x & 1) && (y & 1)) cnt[1]++;
		if ((x & 1) && !(y & 1)) cnt[2]++;
		if (!(x & 1) && !(y & 1)) cnt[3]++;
	}
	cout << n * m - abs(cnt[0] + cnt[3] - cnt[1] - cnt[2]) << endl;
	return 0;
}
