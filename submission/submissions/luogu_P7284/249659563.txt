#include<bits/stdc++.h>
#define file(a) freopen(#a".in", "r", stdin), freopen(#a".out", "w", stdout)
//#define int long long
using namespace std;
const int dx[] = { 0,0,1,-1 }, dy[] = { -1,1,0,0 };
int idx[2000 + 5][2000 + 5][4];
short type[2000 + 5][2000 + 5];
char mp[2000 + 5][2000 + 5];
int srcx, srcy, dstx, dsty;
int get_dir(int i, int j) {
	switch (mp[i][j]) {
	case '<': return 0;
	case '>': return 1;
	case 'v': return 2;
	case '^': return 3;
	case 'o': srcx = i; srcy = j; return 4;
	case 'x': dstx = i; dsty = j; return 5;
	default:  return 6;
	}
}
int n, m;
int dis[2000 + 5][2000 + 5];
bool vis[2000 + 5][2000 + 5];
pair<int, int> from[2000 + 5][2000 + 5];
bool inside(int nx, int ny) {
	if (nx > n || ny > m || nx < 1 || ny < 1) return 0;
	return 1;
}
int32_t main() {
//	file(patkice22);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m;
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= m; ++j)
			cin >> mp[i][j], type[i][j] = get_dir(i, j);
	for (int i = 1, idk = 0; i <= n; ++i)
		for (int j = 1; j <= m; ++j)
			for (int k = 0; k < 4; ++k)
				idx[i][j][k] = ++idk;
	memset(dis, 0x3f, sizeof dis);
	priority_queue<tuple<int, int, int>, vector< tuple<int, int, int>>, greater< tuple<int, int, int>>>q;
	for (int i = 0; i < 4; ++i) {
		int x = srcx + dx[i], y = srcy + dy[i];
		if (inside(x, y))
			q.emplace(0, x, y), dis[x][y] = 0,
			from[x][y] = { srcx,srcy };
	}
	vis[srcx][srcy] = 1; dis[srcx][srcy] = 0;
	while (!q.empty()) {
		auto [d, x, y] = q.top(); q.pop();
		if (vis[x][y]) continue; vis[x][y] = 1;
		if (x == dstx && y == dsty) break;
		for (int i = 0; i < 4; ++i) {
			int nx = x + dx[i], ny = y + dy[i];
			if (inside(nx, ny) && dis[nx][ny] > d + (i != type[x][y]))
				dis[nx][ny] = d + (i != type[x][y]), q.emplace(dis[nx][ny], nx, ny),
				from[nx][ny] = { x,y };
		}
	}
	cout << dis[dstx][dsty] << endl;
	int curx = dstx, cury = dsty;
	while (from[curx][cury] != make_pair(srcx, srcy)) {
		auto [nxtx, nxty] = from[curx][cury];
		if (nxtx + dx[0] == curx && nxty + dy[0] == cury) {
			mp[nxtx][nxty] = '<';
		}
		if (nxtx + dx[1] == curx && nxty + dy[1] == cury) {
			mp[nxtx][nxty] = '>';
		}
		if (nxtx + dx[2] == curx && nxty + dy[2] == cury) {
			mp[nxtx][nxty] = 'v';
		}
		if (nxtx + dx[3] == curx && nxty + dy[3] == cury) {
			mp[nxtx][nxty] = '^';
		}
		tie(curx, cury) = from[curx][cury];
	}
	for (int i = 1; i <= n; cout << endl, ++i)
		for (int j = 1; j <= m; ++j)
			cout << mp[i][j];
	return 0;
}
