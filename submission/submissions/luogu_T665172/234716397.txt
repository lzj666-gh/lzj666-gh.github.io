using namespace std;
constexpr int mod = 1e9 + 7;
int n, a[100000 + 5], ans;
vector<int>v[100000 + 5];
int qpow(int b) {
	int a = 2, res = 1;
	while (b) {
		if (b & 1) res = (res * a) % mod;
		a = (a * a) % mod; b >>= 1;
	}
	return res;
}
int calc(vector<int>& v) {
	int s = 0, t = 0;
	for (int& x : v) s |= x, t += x;
	return s % mod * qpow(v.size() - 1) % mod - t % mod;
}
int32_t main() {
	// file(climb);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n;
	for (int i = 1; i <= n; ++i)
		cin >> a[i], v[i].push_back(a[i]);
	for (int i = 2; i <= n; ++i) {
		int p; cin >> p;
		v[p].push_back(a[i]);
	}
	for (int i = 2; i <= n; ++i) {
		ans += calc(v[i]) * qpow(n - v[i].size() - 1);
		ans %= mod;
	}
	ans += calc(v[1]) * qpow(n - v[1].size()); ans %= mod;
	v[1].erase(v[1].begin());
	ans -= calc(v[1]) * qpow(n - v[1].size() - 1); ans %= mod;
	cout << (ans + mod) % mod << endl;
	return 0;
}
