#include<bits/stdc++.h>
#define int long long

#define rall(a) (a).rbegin(),a.rend()

using namespace std;
constexpr int t = 400;
int n, m, a[200000 + 5], bel[200000 + 5], rst[500], lft[500];
struct Block {
	int blk[500], a[200000 + 5];
	void add(int x, int k) {
		blk[bel[x]] += k;
		a[x] += k;
	}
	int query(int l, int r) {
		int sum = 0;
		if (bel[l] == bel[r]) {
			for (int i = l; i <= r; ++i)
				sum += a[i];
			return sum;
		}
		for (int i = l; i <= rst[bel[l]]; ++i)
			sum += a[i];
		for (int i = lft[bel[r]]; i <= r; ++i)
			sum += a[i];
		for (int i = bel[l] + 1; i < bel[r]; ++i)
			sum += blk[i];
		return sum;
	}
}nor, tim;
struct Fenw_Tree {
#define lb(x) (x & ~x + 1)
	int s[1000 + 10], c[1000 + 10];
	int query1(int x) {
		int ans = 0;
		for (; x; x -= lb(x)) {
			ans += s[x];
		}
		return ans;
	}
	int query2(int x) {
		int ans = 0;
		for (; x; x -= lb(x)) {
			ans += c[x];
		}
		return ans;
	}
	void add1(int x, int k) {
		for (; x <= n; x += lb(x)) {
			s[x] += k;
		}
	}
	void add2(int x, int k) {
		for (; x <= n; x += lb(x)) {
			c[x] += k;
		}
	}
	void add(int x, int y, int k) {
		add1(x, k), add1(y + 1, ~k + 1);
		add2(x, k * (x - 1)), add2(y + 1, (~k + 1) * y);
	}
	int query(int x, int y) {
		return y * query1(y) - query2(y) - ((x - 1) * query1(x - 1) - query2(x - 1));
	}
}tr[t + 1];
int32_t main() {
	//file(scarlet);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m;
	for (int i = 1; i <= n; ++i)
		cin >> a[i];
	for (int i = 1; i <= n; ++i)
		bel[i] = i / t, rst[i / t] = i,
		((lft[i / t]) || (lft[i / t] = i));
	for (int i = 1; i <= n; ++i)
		nor.add(i, a[i] - a[i - 1]),
		tim.add(i, (a[i] - a[i - 1]) * i);
	while (m--) {
		int opt, x, y, k; cin >> opt >> x >> y;
		if (opt == 1) {
			cin >> k;
			if (y >= x - 1) nor.add(1, k), tim.add(1, k);
			else if (x > t)
				for (int i = 1; i <= n; i += x)
					nor.add(i, k), nor.add(i + y + 2, -k),
					tim.add(i, i * k), tim.add(i + y + 2, (i + y + 2) * -k);
			else tr[x].add(1, y + 1, k);
		}
		else {
			long long ans = 0; --x;
			for (int i = 1; i <= t; ++i)
				ans += tr[i].query(1, n) * (y / i - x / i) + tr[i].query(1, y % i) - tr[i].query(1, x % i);
			ans += nor.query(1, y) * (y + 1) - tim.query(1, y) - nor.query(1, x) * (x + 1) + tim.query(1, x);
			cout << ans << endl;
		}
		//for (int i = 1; i <= n; ++i) cout << nor.query(i, i) << " \n"[i == n];
	}
	return 0;
}
