#include<bits/stdc++.h>

#define file(name) freopen(#name".in", "r", stdin), freopen(#name".out", "w", stdout);
using namespace std;
constexpr int mod = 1e9 + 7;
int s[300000 + 5];
int qpow(int a, int b) {
	int res = 1;
	while (b) {
		if (b & 1) res = res * a % mod;
		a = a * a % mod; b >>= 1;
	}
	return res;
}
int32_t main() {
	// file(c);
	cin.tie(0)->sync_with_stdio(false);
	int n, m; cin >> n >> m;
	for (int i = 1; i <= n; ++i) cin >> s[i];
	if (n <= 5e4 && 1) {
		static int cntl[50000 + 5], cntr[50000 + 5], cnt[50000 + 5];
		for (int i = 1; i <= n; ++i) {
			cntl[i] = 1;
			for (int j = 1; j < i; ++j)
				if ((s[i] & s[j]) == s[j])
					cntl[i] += cntl[j], cntl[i] %= mod;
		}
		for (int i = n; i >= 1; --i) {
			cntr[i] = cnt[i] = 1;
			for (int j = n; j > i; --j)
				if ((s[i] & s[j]) == s[j])
					cntr[i] += cntr[j], cntr[i] %= mod;
			for (int j = n; j > i; --j)
				if ((s[i] & s[j]) == s[j] && s[i] != s[j])
					cnt[i] += cntr[j], cnt[i] %= mod;
		}
		int ans = 0;
		for (int i = 1; i <= n; ++i) ans = (ans + cntl[i] * cnt[i] % mod) % mod;
		cout << ans << endl;
		return 0;
	}
	if (m <= 10 && 1) {
		static int cntl[300000 + 5], cntr[300000 + 5], cnt[300000 + 5], cntll[1024 + 5], cntrr[1024 + 5];
		int lim = 1 << m;
		for (int i = 1; i <= n; ++i) {
			cntl[i] = 1;
			for (int j = 0; j < lim; ++j)
				if ((j & s[i]) == j)
					cntl[i] += cntll[j], cntl[i] %= mod;
			cntll[s[i]] += cntl[i]; cntll[s[i]] %= mod;
		}
		for (int i = n; i >= 1; --i) {
			cntr[i] = cnt[i] = 1;
			for (int j = 0; j < lim; ++j)
				if ((j & s[i]) == j)
					cntr[i] += cntrr[j], cntr[i] %= mod;
			for (int j = 0; j < lim; ++j)
				if ((j & s[i]) == j && s[i] != j)
					cnt[i] += cntrr[j], cnt[i] %= mod;
			cntrr[s[i]] += cntr[i]; cntrr[s[i]] %= mod;
		}
		int ans = 0;
		for (int i = 1; i <= n; ++i) ans = (ans + cntl[i] * cnt[i] % mod) % mod;
		cout << ans << endl;
		return 0;
	}
	cout << 0 << endl;
	return 0;
}
