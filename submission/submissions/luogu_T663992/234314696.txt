#include<bits/stdc++.h>

using namespace std;
constexpr int mod = 998244353;
int qpow(int a, int b) {
	int res = 1;
	while (b) {
		if (b & 1) res = res * a % mod;
		a = a * a % mod; b >>= 1;
	}
	return res;
}
int cnt[50 + 5], idx[50 + 5], inv[10000 + 5];
int n, p, m;
int C(int n, int m) {
	int res = 1;
	for (int i = m + 1; i <= n; ++i) res = res * i % mod;
	for (int i = 1; i <= n - m; ++i) res = res * inv[i] % mod;
	return res;
}
int cntt[50 + 5];
int dfs(int x, int k) {
	if (k == 0) {
		int res = 0;
		for (int i = 0; i < x; ++i) res = (res + cntt[i] * i) % p;
		if (res) return 0;
		//for (int i = 0; i < x; ++i) cerr << cntt[i] << ' '; cerr << endl;
		res = 1;
		for (int i = 0; i < x; ++i) res = res * C(cnt[i] + cntt[i], cnt[i]) % mod;
		return res;
	}
	if (x >= p) return 0;
	int res = 0;
	for (int i = 0; i <= k; ++i) {
		cntt[x] = i;
		res = (res + dfs(x + 1, k - i)) % mod;
	}
	return res;
}
int32_t main() {
//	file(probability);
	(cin >> n >> p >> m).tie(0)->sync_with_stdio(false);
	for (int i = 1; i <= m; ++i) inv[i] = qpow(i, mod - 2);
	int k = 1; bool flg = 1;
	for (int i = 1; i <= n; ++i) {
		if (cnt[k] && flg) {
			int cir = i - idx[k]; flg = 0;
			int tim = (n - i) / cir;
			for (int t = 0; t < cir; ++t) {
				cnt[k] += tim;
				k = k * 10 % p;
			}
			i += tim * cir - 1;
			continue;
		}
		cnt[k]++;
		idx[k] = i;
		k = k * 10 % p;
	}
	for (int i = 0; i < p; ++i) cnt[i]--;
	if (m <= 20) {
		int res = 0;
		for (int k = 0; k <= m; ++k) {
			//cerr << k << ":" << endl;
			res = (res + dfs(0, k)) % mod;
			cout << res << ' ';
		}
		return 0;
	}
	int res = 0;
	for (int k = 0; k <= m; ++k) {
		//cerr << k << ":" << endl;
		res = (res + C(cnt[0] + k, cnt[0])) % mod;
		if (k == p) {
			res = (res + C(cnt[1] + k, cnt[1])) % mod;
		}
		cout << res << ' ';
	}
	return 0;
}
