#include<bits/stdc++.h>
using namespace std;
int main() {
	cin.tie(0)->sync_with_stdio(false);
	int X, T;
	cin >> X >> T;
	if (X == 1) {
		while (T--) {
			int n; cin >> n;
			vector<int>a(n + 1), b(n + 1);
			for (int i = 1; i <= n; ++i) {
				if (i % 2 == 0) a[i] = i - 1;
				else a[i] = n - i + 1;
			}
			iota(b.begin(), b.end(), 0);
			if (is_permutation(a.begin(), a.end(), b.begin())) {
				cout << 2 << ' ';
				for (int i = 1; i <= n; ++i) cout << (a[i] += a[i - 1]) % n << " \n"[i == n];
			}
			else cout << 0 << endl;
		}
	}
	return 0;
}
