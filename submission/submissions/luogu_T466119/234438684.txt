using namespace std;
constexpr int maxi = 100;
int qpow(int a, int b, int p) {
	int res = 1;
	while (b) {
		if (b & 1) res = res * a % p;
		a = a * a % p; b >>= 1;
	}
	return res;
}
int dp[2][maxi + 5][2];
int32_t main() {
	// file(b);
	cin.tie(0)->sync_with_stdio(false);
	int n, m, p; cin >> n >> m >> p;
	if (n > maxi) { cout << 0 << endl; return 0; }
	vector<int>d;
	for (int i = 1; i * i <= m; ++i)
		if (m % i == 0) {
			d.push_back(i);
			if (i * i < m) d.push_back(m / i);
		}
	int ans = 0, facn = 1, ifacn;
	for (int i = 1; i <= n; ++i) facn = facn * i % p;
	ifacn = qpow(facn, p - 2, p);
	for (int& k : d) {
		if (--k == 0) continue;
		vector<pair<int, int>> fd;
		for (int i = 2; i * i <= k; ++i)
			if (k % i == 0) {
				int cnt = 0;
				while (k % i == 0) cnt++, k /= i;
				fd.emplace_back(i, cnt);
			}
		if (k > 1) fd.emplace_back(k, 1);
		int D = 1;
		for (auto& f : fd) {
			D *= (f.second + 1);
			if (D > maxi) break;
		}
		if (n > D) continue;
		memset(dp[0], 0, sizeof dp[0]);
		int cur = 0; dp[cur][1][0] = 1;
		for (auto& f : fd) {
			int t = f.second; cur ^= 1;
			memset(dp[cur], 0, sizeof dp[cur]);
			for (int j = 1; j <= maxi; ++j)
				for (int pp : {0, 1}) {
					if (dp[j][pp] == 0) continue;
					int cj = j * (t + 1), cp = pp;
					if (cj <= maxi)
						dp[cur][cj][cp] = (dp[cur][cj][cp] + dp[cur ^ 1][j][pp]) % p;
					cj = j * t; cp = pp ^ 1;
					if (cj <= maxi)
						dp[cur][cj][cp] = (dp[cur][cj][cp] + dp[cur ^ 1][j][pp] + dp[cur ^ 1][j][pp]) % p;
					cj = j * (t - 1); cp = pp;
					if (t - 1 && cj <= maxi)
						dp[cur][cj][cp] = (dp[cur][cj][cp] + dp[cur ^ 1][j][pp]) % p;
				}
		}
		for (int j = 1; j <= maxi; ++j) {
			if (j < n) continue; int c = ifacn;
			for (int i = 0; i < n; ++i) c = c * (j - i) % p;
			ans = (ans + (dp[cur][j][0] - dp[cur][j][1]) * c % p) % p;
		}
	}
	ans = (ans % p + p) % p;
	cout << ans << endl;
	return 0;
}
