using ll = long long;
int n, m; ll p;
struct Node {
	int pos, buy, price, cought;
	friend bool operator<(const Node& a, const Node& b) {
		return a.pos == b.pos ? a.buy > b.buy:a.pos < b.pos;
	}
	friend ll operator-(const Node& a, const Node& b) {
		return (a.pos - b.pos) * p;
	}
}a[1000000 + 5];
struct Seg_Tree {
#define lc(p) (tree[p].ls == -1 ? tree[p].ls = ++idx : tree[p].ls)
#define rc(p) (tree[p].rs == -1 ? tree[p].rs = ++idx : tree[p].rs)
#define mid (l + r >> 1)
#define lson lc(p), l, mid
#define rson rc(p), mid + 1, r
	struct Node {
		int ls, rs; ll sum, cnt;
		Node() :ls(-1), rs(-1), sum(0), cnt(0) {}
	}tree[10000000 + 5];
	int idx;
	void pushup(int p) {
		tree[p].sum = 0; tree[p].cnt = 0;
		if (~tree[p].ls) {
			tree[p].sum += tree[tree[p].ls].sum;
			tree[p].cnt += tree[tree[p].ls].cnt;
		}
		if (~tree[p].rs) {
			tree[p].sum += tree[tree[p].rs].sum;
			tree[p].cnt += tree[tree[p].rs].cnt;
		}
	}
	void add(int p, int l, int r, const int& x, const int& k) {
		tree[p].sum += 1ll * x * k; tree[p].cnt += k;
		if (l == r) return;
		if (x <= mid) add(lson, x, k);
		else add(rson, x, k);
	}
	ll query(int p, int l, int r, ll k) {
		//cerr << "query [" << l << ',' << r << "]; k = " << k << " and tree[p].cnt = " << tree[p].cnt << endl;
		if (tree[p].cnt <= k) return tree[p].sum;
		if (l == r) return k * l;
		ll res = 0;
		if (~tree[p].rs) {
			if (tree[tree[p].rs].cnt <= k)
				res += tree[tree[p].rs].sum, k -= tree[tree[p].rs].cnt;
			else return query(rson, k);
		}
		if (~tree[p].ls) {
			return res + query(lson, k);
		}
		return res;
	}
	Seg_Tree() :idx(1), tree() {}
}t;
