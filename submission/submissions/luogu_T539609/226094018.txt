#include<bits/stdc++.h>
#define int long long
using namespace std;
constexpr int mod = 998244353;
int n, m, a[10005], sum[50005], dp[50005];
int32_t main(int32_t argc, char** argv) {
	//if (argc >= 2) freopen(argv[1], "r", stdin);
	//else freopen("b.in", "r", stdin);
	//if (argc >= 3) freopen(argv[2], "w", stdout);
	//else freopen("b.out", "w", stdout);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m; int magic = log2(m) * sqrt(m);
	for (int i = 1; i <= n; ++i) cin >> a[i];
	sort(a + 1, a + n + 1);
	for (int i = n, j = 1; i >= 1; --i, ++j)
		for (int k = a[i] - 1; k >= a[i - 1]; --k)
			sum[k] = j;
	memset(dp, 0x3f, sizeof dp);
	for (int i = 1; i <= m; ++i) dp[i] = sum[i];
	for (int i = 2; i <= n; ++i) {
		for (int j = m; j >= 1; --j) {
			dp[j] = dp[j] + sum[0];
			for (int k = 1; k <= min(j, m / j + magic); ++k)
				dp[j] = min(dp[j], dp[j - k] + sum[k]);
		}
	}
	int ans = dp[m] % mod;
	for (int i = 1; i < n; ++i) ans *= i, ans %= mod;
	cout << ans << endl;
	return 0;
}
