#include<bits/stdc++.h>
using namespace std;
int exgcd(int a, int b, int& x, int& y) {
    if (b == 0) { x = 1; y = 0; return a; }
    int k = exgcd(b, a % b, y, x);
    y -= a / b * x;
    return k;
}
inline int inv(int a, int p) {
    int x, y; exgcd(a, p, x, y);
    return (x % p + p) % p;
}
int a, b;
int main() {
    cin.tie(0)->sync_with_stdio(false);
    cin >> a >> b;
    cout << inv(a, b) << endl;
    return 0;
}
