#include<bits/stdc++.h>
#define int long long
using namespace std;
int n, p, q, r, a[100000 + 5];
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> p >> q >> r;
	int sum = 0;
	for (int i = 1; i <= n; ++i) cin >> a[i], sum += a[i];
	int ans = 0;
	sort(a + 1, a + n + 1);
	for (int i = 1; i < n; ++i) {
		if (sum > 0) {
			cout << ans << endl;
			return 0;
		}
		ans += min(p * min(-sum, -a[i]), q);
		sum -= a[i];
	}
	if (sum < 0) ans += -sum * p;
	cout << ans << endl;
	return 0;
}
