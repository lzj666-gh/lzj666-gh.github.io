#include<bits/stdc++.h>
#define int long long
#define file(name) freopen(#name".in", "r", stdin), freopen(#name".out", "w", stdout);
using namespace std;
int x, y, z, a, n;
int dp[100000 + 5];
vector<int>c, m;
int32_t main() {
//	file(tourist);
	cin.tie(0)->sync_with_stdio(false);
	cin >> x >> y >> z >> a >> n; c.resize(n); m.resize(n);
	for (int i = 0; i < n; ++i) cin >> c[i] >> m[i];
	memset(dp, 0xcf, sizeof dp);
	int beg = lower_bound(c.begin(), c.end(), x) - c.begin();
	if (c[beg] == x) dp[beg] = m[beg];
	else {
		c.insert(c.begin() + beg, x);
		m.insert(m.begin() + beg, a);
		dp[beg] = 0;
	}
	int siz = 2e8 / (lower_bound(c.begin(), c.end(), y) - lower_bound(c.begin(), c.end(), x));
	int i; deque<int>q; q.push_back(beg);
	for (i = beg + 1; i <= n && c[i] <= y; ++i) {
		for (int& j : q)
			dp[i] = max(dp[i], dp[j] - (c[i] - 1 - c[j]) / z * a + m[i] - a);
		while (!q.empty() && dp[i] >= dp[q.back()]) q.pop_back();
		q.push_back(i);
		if (q.size() > siz) q.pop_front();
	}
	i--; int ans = LLONG_MIN;
	while (i >= beg) {
		if (c[i] == y) ans = max(ans, dp[i]);
		else ans = max(ans, dp[i] - a - (y - 1 - c[i]) / z * a);
		--i;
	}
	cout << ans << endl;
	return 0;
}
