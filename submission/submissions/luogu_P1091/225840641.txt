#include<bits/stdc++.h>
using namespace std;
int n, a[1000], dp[2][1000];
signed main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n;
	for (int i = 1; i <= n; ++i) cin >> a[i], dp[0][i] = dp[1][i] = 1;
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j < i; ++j)
			if (a[j] < a[i])
				dp[0][i] = max(dp[0][i], dp[0][j] + 1);
	for (int i = n; i >= 1; --i)
		for (int j = n; j > i; --j)
			if (a[j] < a[i])
				dp[1][i] = max(dp[1][i], dp[1][j] + 1);
	int ans = 0;
	for (int i = 1; i <= n; ++i)
		ans = max(ans, dp[0][i] + dp[1][i] - 1);
	cout << n - ans << endl;
	return 0;
}
