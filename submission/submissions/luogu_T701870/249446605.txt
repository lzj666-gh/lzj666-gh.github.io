#include<bits/stdc++.h>
#define file(a) freopen(#a".in", "r", stdin), freopen(#a".out", "w", stdout)
//#define int long long
using namespace std;
constexpr int mod = 1e9 + 7;
int32_t main() {
	//	file(olympics);
	cin.tie(0)->sync_with_stdio(false);
	int n, m; cin >> n >> m;
	vector<vector<int>>sum(n + 1, vector<int>(m + 1));
	for (int i = 1; i <= n; ++i) {
		string s; cin >> s; s = ' ' + s;
		for (int j = 1; j <= m; ++j)
			sum[i][j] = s[j] == '1';
	}
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= m; ++j)
			sum[i][j] += sum[i - 1][j] + sum[i][j - 1] - sum[i - 1][j - 1];
	function<int(int, int, int, int)>get = [&](int x1, int y1, int x2, int y2) { return sum[x2][y2] - sum[x1 - 1][y2] - sum[x2][y1 - 1] + sum[x1 - 1][y1 - 1]; };
	if (n < m) {
		vector<vector<int>>dp[2];
		dp[0] = dp[1] = vector<vector<int>>(n + 1, vector<int>(n + 1, 0));
		int ans = 0;
		for (int k = 1; k <= m; ++k) {
			for (int i = n; i >= 1; --i) {
				int mx = 0;
				for (int j = 1; j <= i; ++j) {
					mx = max(mx, dp[k & 1 ^ 1][i][j]);
					if (get(j, k, i, k) == i - j + 1)
						ans = max(ans, dp[k & 1][i][j] = mx + i - j + 1);
					else {
						fill(dp[k & 1][i].begin() + j, dp[k & 1][i].begin() + i + 1, 0);
						break;
					}
				}
			}
		}
		cout << ans << endl;
	}
	else {
		vector<vector<int>>dp[2];
		dp[0] = dp[1] = vector<vector<int>>(m + 1, vector<int>(m + 1, 0));
		int ans = 0;
		for (int k = 1; k <= n; ++k) {
			for (int i = 1; i <= m; ++i) {
				int mx = 0;
				for (int j = i; j <= m; ++j) {
					mx = max(mx, dp[k & 1 ^ 1][i][j]);
					if (get(k, i, k, j) == j - i + 1)
						ans = max(ans, dp[k & 1][i][j] = mx + j - i + 1);
					else {
						fill(dp[k & 1][i].begin() + j, dp[k & 1][i].end(), 0);
						break;
					}
				}
			}
		}
		cout << ans << endl;
	}
	return 0;
}
