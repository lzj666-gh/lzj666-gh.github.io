#include<bits/stdc++.h>
#define int long long
using namespace std;
int qpow(int a, int b, int p) {
	int res = 1;
	while (b) {
		if (b & 1) (res *= a) %= p;
		(a *= a) %= p; b >>= 1;
	}
	return res;
}
int a, b, p;
signed main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> a >> b >> p;
	cout << a << '^' << b << " mod " << p << '=' << qpow(a, b, p) << endl;
	return 0;
}
