struct Seg_Tree {
#define lc(p) (p << 1)
#define rc(p) (p << 1 | 1)
#define mid (l + r >> 1)
#define lson lc(p), l, mid
#define rson rc(p), mid + 1, r
	struct Node {
		int min;
		Node() : min(1e18) {}
	}tree[100000 << 2 | 3];
	void pushup(int p) {
		tree[p].min = min(tree[lc(p)].min, tree[rc(p)].min);
	}
	void update(int p, int l, int r, int x, int k) {
		if (l == r) {
			tree[p].min = k;
			return;
		}
		if (x <= mid) update(lson, x, k);
		else update(rson, x, k);
		pushup(p);
	}
	int query(int p, int l, int r, int x, int y) {
		if (x <= l && r <= y) return tree[p].min;
		if (r < x || y < l || r < l) return 1e18;
		return min(query(lson, x, y), query(rson, x, y));
	}
	void clear(int n) {
		fill(tree + 1, tree + (n << 2) + 1, Node());
	}
#undef mid
