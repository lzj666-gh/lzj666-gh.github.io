#include<bits/stdc++.h>
using namespace std;
constexpr int N = 1e4, M = 1e5;
struct Edge {
	int v, nxt;
	Edge(int v = 0, int nxt = 0) : v(v), nxt(nxt) {}
}edge[M];
int r[N], head[N], idx;
void add(int u, int v) {
	edge[++idx] = { v,head[u] };
	head[u] = idx; ++r[v];
}
int n, m, T[N], t[N];
int main(int argc, char** argv, char** env) {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m;
	for (int i = 0; i < n; ++i) cin >> T[i];
	queue<int>q;
	for (int i = 0, a, b; i < m; ++i) cin >> a >> b, add(--a, --b);
	for (int i = 0; i < n; ++i) {
		if (!r[i])
			q.emplace(i);
	}
	int ans = 0;
	while (!q.empty()) {
		int x = q.front(); q.pop(); ans = max(ans, t[x] + T[x]);
		for (int i = head[x]; i; i = edge[i].nxt) {
			t[edge[i].v] = max(t[edge[i].v], t[x] + T[x]);
			if (!--r[edge[i].v])
				q.emplace(edge[i].v);
		} 
	}
	cout << ans << endl;
//	for (int i = 0; i < n; ++i) cout << t[i] << " \n"[i == n - 1];
	return 0;
}
