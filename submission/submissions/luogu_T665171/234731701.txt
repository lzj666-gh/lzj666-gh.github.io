#include<bits/stdc++.h>
#define int long long
using namespace std;
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	int a, b, c, d; cin >> a >> b >> c >> d;
	auto check1 = [&a, &b, &c, &d](int i, int j, int k)->int {
		int p = (a - (i + j / 2 + k / 2)) * 4;
		p = max(p, (b - (i / 2 + j + k / 4)) * 2);
		p = max(p, (c - (i / 2 + j / 4 + k)) * 2);
		p = max(p, (d - (i / 4 + j / 2 + k / 2)));
		if (p < 0) return i + j + k;
		else return i + j + k + p;
		};
	auto check2 = [&a, &b, &c, &d, &check1](int i, int j)->int {
		int l = 0, r = c, mid1, mid2;
		while (l < r - 10) {
			mid1 = l + (r - l + 1) / 3;
			mid2 = r - (r - l + 1) / 3;
			if (check1(i, j, mid1) > check1(i, j, mid2)) l = mid1;
			else r = mid2;
		}
		int ans = a + b + c + d;
		for (int k = l; k <= r; ++k)
			ans = min(ans, check1(i, j, k));
		return ans;
		};
	auto check3 = [&a, &b, &c, &d, &check2](int i)->int {
		int l = 0, r = c, mid1, mid2;
		while (l < r - 100) {
			mid1 = l + (r - l + 1) / 3;
			mid2 = r - (r - l + 1) / 3;
			if (check2(i, mid1) > check2(i, mid2)) l = mid1;
			else r = mid2;
		}
		int ans = a + b + c + d;
		for (int k = l; k <= r; ++k)
			ans = min(ans, check2(i, k));
		return ans;
		};
	auto check4 = [&a, &b, &c, &d, &check3]()->int {
		int l = 0, r = c, mid1, mid2;
		while (l < r - 1000) {
			mid1 = l + (r - l + 1) / 3;
			mid2 = r - (r - l + 1) / 3;
			if (check3(mid1) > check3(mid2)) l = mid1;
			else r = mid2;
		}
		int ans = a + b + c + d;
		for (int k = l; k <= r; ++k)
			ans = min(ans, check3(k));
		return ans;
		};
	cout << check4() << endl;
	return 0;
}
