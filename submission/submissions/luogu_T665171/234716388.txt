#include<bits/stdc++.h>
#define int long long
#define file(name) freopen(#name".in", "r", stdin), freopen(#name".out", "w", stdout);
using namespace std;
int32_t main() {
	// file(light);
	cin.tie(0)->sync_with_stdio(false);
	int a, b, c, d;
	cin >> a >> b >> c >> d;
	int ans = a + b + c + d;
	if (a == max({ a,b,c,d }) && b * c * d <= 1e8 && 1) {
		for (int i = 0; i <= b; ++i)
			for (int j = 0; j <= c; ++j)
				for (int k = 0; k <= d; ++k) {
					// [p, i]
					// [j, k]
					int p = a - (i / 2 + j / 2 + k / 4);
					p = max(p, (b - (i + j / 4 + k / 2)) * 2);
					p = max(p, (c - (i / 4 + j + k / 2)) * 2);
					p = max(p, (d - (i / 2 + j / 2 + k)) * 4);
					if (p < 0) ans = min(ans, i + j + k);
					else ans = min(ans, i + j + k + p);
				}
	}
	else if (b == max({ a,b,c,d }) && a * c * d <= 1e8 && 1) {
		for (int i = 0; i <= a; ++i)
			for (int j = 0; j <= c; ++j)
				for (int k = 0; k <= d; ++k) {
					// [i, p]
					// [j, k]
					int p = (a - (i + j / 2 + k / 4)) * 2;
					p = max(p, (b - (i / 2 + j / 4 + k / 2)));
					p = max(p, (c - (i / 2 + j + k / 2)) * 4);
					p = max(p, (d - (i / 4 + j / 2 + k)) * 2);
					if (p < 0) ans = min(ans, i + j + k);
					else ans = min(ans, i + j + k + p);
				}
	}
	else if (c == max({ a,b,c,d }) && a * b * d <= 1e8 && 1) {
		for (int i = 0; i <= a; ++i)
			for (int j = 0; j <= b; ++j)
				for (int k = 0; k <= d; ++k) {
					// [i, j]
					// [p, k]
					int p = (a - (i + j / 2 + k / 4)) * 2;
					p = max(p, (b - (i / 2 + j + k / 2)) * 4);
					p = max(p, (c - (i / 2 + j / 4 + k / 2)));
					p = max(p, (d - (i / 4 + j / 2 + k)) * 2);
					if (p < 0) ans = min(ans, i + j + k);
					else ans = min(ans, i + j + k + p);
				}
	}
	else if (d == max({ a,b,c,d }) && a * b * c <= 1e8 && 1) {
		for (int i = 0; i <= a; ++i)
			for (int j = 0; j <= b; ++j)
				for (int k = 0; k <= c; ++k) {
					// [i, j]
					// [k, p]
					int p = (a - (i + j / 2 + k / 2)) * 4;
					p = max(p, (b - (i / 2 + j + k / 4)) * 2);
					p = max(p, (c - (i / 2 + j / 4 + k)) * 2);
					p = max(p, (d - (i / 4 + j / 2 + k / 2)));
					if (p < 0) ans = min(ans, i + j + k);
					else ans = min(ans, i + j + k + p);
				}
	}
	else {
		auto check = [&a, &b, &c, &d](int i, int j, int k)->int {
			int p = (a - (i + j / 2 + k / 2)) * 4;
			p = max(p, (b - (i / 2 + j + k / 4)) * 2);
			p = max(p, (c - (i / 2 + j / 4 + k)) * 2);
			p = max(p, (d - (i / 4 + j / 2 + k / 2)));
			if (p < 0) return i + j + k;
			else return i + j + k + p;
			};
		for (int i = 0; i <= a; ++i)
			for (int j = 0; j <= b; ++j) {
				// 通过观察，我们发现 k+p 具有大致单谷性，于是我们大胆推测可以三分
				int l = 0, r = c, mid1, mid2;
				while (l < r - 1) {
					mid1 = l + (r - l + 1) / 3;
					mid2 = r - (r - l + 1) / 3;
					if (check(i, j, mid1) > check(i, j, mid2)) l = mid1;
					else r = mid2;
				}
				for (int k = max(0ll, l - 5), lim = min(c, r + 5); k <= lim; ++k)
					ans = min(ans, check(i, j, k));
			}
	}
	cout << ans << endl;
	return 0;
}
