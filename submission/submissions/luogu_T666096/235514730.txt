#include<bits/stdc++.h>
using namespace std;
int T, n, m;
struct Query {
	int l, r;
	friend bool operator<(const Query& a, const Query& b) {
		return a.l == b.l ? a.r < b.r : a.l < b.l;
	}
}q[40000 + 5];
int c[2][40000 + 5];
signed main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> T;
	while (T--) {
		cin >> n >> m; int l = m, r = 1;
		for (int i = 1; i <= n; ++i)
			cin >> q[i].l >> q[i].r,
			c[0][q[i].l]++, c[0][q[i].r]--,
			c[1][q[i].l]++, c[1][q[i].r + 1]--,
			l = min(l, q[i].l), r = max(r, q[i].r);
		bool flg = n == 1;
		for (int i = l; i <= r; ++i)
			if ((c[1][i] += c[1][i - 1]) == n)
				flg = 1;
		long long ans = 3;
		for (int i = l; i < r; ++i)
			if ((c[0][i] += c[0][i - 1]) == 0)
				ans = 2;
		if (flg) cout << "1145141919810" << endl;
		else cout << ans << endl;
		for (int i = l; i <= r + 1; ++i) c[0][i] = c[1][i] = 0;
	}
	return 0;
}
