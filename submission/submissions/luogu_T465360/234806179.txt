#include<bits/stdc++.h>
#define int long long
#define file(name) freopen(#name".in", "r", stdin), freopen(#name".out", "w", stdout);
using namespace std;
int dis[305][305], n;
bool need[305][305];
int32_t main() {
//	file(road);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n;
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= n; ++j)
			cin >> dis[i][j];
	memset(need, 1, sizeof need);
	for (int k = 1; k <= n; ++k)
		for (int i = 1; i <= n; ++i)
			for (int j = 1; j <= n; ++j) {
				if (i != k && j != k && i != j && dis[i][k] + dis[k][j] == dis[i][j])
					need[i][j] = 0;
				if (dis[i][k] + dis[k][j] < dis[i][j]) {
					cout << -1 << endl;
					return 0;
				}
			}
	int ans = 0;
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j < i; ++j)
			if (need[i][j])
				ans += dis[i][j];
	cout << ans << endl;
	return 0;
}
