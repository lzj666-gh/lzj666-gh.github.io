#include<bits/stdc++.h>
using namespace std;
int a, b, cnt;
void dfs(long long k) {
	if (k >= a) cnt++;
	if (k * 10 > b) return;
	int now = k % 10;
	k *= 10;
	for (int i = 0; i < 10; ++i) {
		if (now - i < 0 ? now - i <= -2 : now - i >= 2) dfs(k + i);
	}
}
int main() {
	ios_base::sync_with_stdio(false);
	cin >> a >> b;
	for (int i = 1; i ^ 10; ++i) dfs(i);
	cout << cnt << endl;
	return 0;
}
