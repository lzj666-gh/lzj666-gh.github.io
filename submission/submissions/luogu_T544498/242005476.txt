#include<bits/stdc++.h>
#define int long long
#define all(a) (a).begin(), (a).end()
#define file(a) { freopen(#a".in", "r", stdin); freopen(#a".out", "w", stdout); }
using namespace std;
string s; int n, m;
int border[2000 + 5];
void prep() {
	for (int i = 1; i < n; ++i) {
		border[i] = border[i - 1];
		while (border[i] && s[i] != s[border[i]])
			border[i] = border[border[i] - 1];
		if (s[i] == s[border[i]]) ++border[i];

	}
}
int dep[2000 + 5], fail[2000 + 5][27];
void build() {
	for (int i = 1; i <= n; ++i)
		dep[i] = dep[border[i - 1]] + 1;
	for (int j = 0; j < 26; ++j)
		fail[0][j] = s[0] - 'a' == j;
	for (int i = 1; i <= n; ++i) {
		for (int j = 0; j < 26; ++j) {
			if (i < n && s[i] - 'a' == j) fail[i][j] = i + 1;
			else fail[i][j] = fail[border[i - 1]][j];
		}
	}
}
int dp[2][2000 + 5];
// dp[i][j]：长度为i的字符串，当前字符串后缀与s的最长匹配长度为j时，能够得到的答案
int32_t main() {
	// file(c);
	cin.tie(0)->sync_with_stdio(false);
	cin >> s >> m; n = s.size();
	prep(); build();
	memset(dp, 0xcf, sizeof dp);
	int inf = 0; swap(inf, dp[0][0]);
	for (int i = 0; i < m; ++i) {
		int cur = i & 1, nxt = cur ^ 1;
		memset(dp[nxt], 0xcf, sizeof dp[nxt]);
		for (int j = 0; j <= n; ++j) {
			if (dp[cur][j] == inf) continue;
			for (int k = 0; k < 26; ++k)
				dp[nxt][fail[j][k]] = max(dp[nxt][fail[j][k]], dp[cur][j] + dep[fail[j][k]]);
		}
	}
	int ans = 0;
	for (int i = 0; i <= n; ++i) ans = max(ans, dp[m & 1][i]);
	cout << ans << endl;
	return 0;
}
