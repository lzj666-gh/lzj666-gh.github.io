#include<bits/stdc++.h>
#include<ext/pb_ds/assoc_container.hpp>
#include<ext/pb_ds/priority_queue.hpp>
#define file(name) freopen(#name".in", "r", stdin), freopen(#name".out", "w", stdout);
using namespace std;
struct Edge {
	int v, w, nxt;
}edge[20'000'000 + 305];
int head[1'000'000 + 30 + 5], idx;
void add(int u, int v, int w) {
	edge[++idx] = { v,w,head[u] };
	head[u] = idx;
}
int n, m, q, a[1'000'000 + 5];
union DIS {
	int dis1[2000 + 30 + 5][2000 + 30 + 5];
	int dis2[30 + 5][1'000'000 + 5];
}d;
bool vis1[2000 + 30 + 5], vis2[1'000'000 + 30 + 5];
void dij1(int s) {
	priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>>q;
	q.emplace(0, s); memset(d.dis1[s], 0x3f, sizeof d.dis1[s]); d.dis1[s][s] = 0;
	while (!q.empty()) {
		int now = q.top().second; q.pop();
		if (vis1[now]) continue;
		vis1[now] = 1;
		for (int i = head[now]; i; i = edge[i].nxt)
			if (edge[i].w + d.dis1[s][now] < d.dis1[s][edge[i].v])
				d.dis1[s][edge[i].v] = edge[i].w + d.dis1[s][now],
				q.emplace(d.dis1[s][edge[i].v], edge[i].v);
	}
	memset(vis1, 0, sizeof vis1);
}
void dij2(int s) {
	__gnu_pbds::priority_queue<pair<int, int>, greater<pair<int, int>>, __gnu_pbds::pairing_heap_tag>q;
	q.push({ 0,s + n + 1 }); memset(d.dis2[s], 0x3f, sizeof d.dis2[s]); d.dis2[s][n + s + 1] = 0;
	while (!q.empty()) {
		int now = q.top().second; q.pop();
		if (vis2[now]) continue;
		vis2[now] = 1;
		for (int i = head[now]; i; i = edge[i].nxt)
			if (edge[i].w + d.dis2[s][now] < d.dis2[s][edge[i].v])
				d.dis2[s][edge[i].v] = edge[i].w + d.dis2[s][now],
				q.push({ d.dis2[s][edge[i].v],edge[i].v });
	}
	memset(vis2, 0, sizeof vis2);
}
int32_t main() {
	// file(andgraph);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m >> q;
	for (int i = 1; i <= n; ++i) {
		cin >> a[i];
		for (int j = 0; j <= 22; ++j)
			if (a[i] & (1ll << j))
				add(i, n + j + 1, a[i]),
				add(n + j + 1, i, a[i]);
	}
	if (n <= 2000 && 1) {
		for (int i = 1; i <= n; ++i) dij1(i);
		while (q--) {
			int s, t; cin >> s >> t;
			if (s == t) {
				cout << 2 * a[s] << endl;
				continue;
			}
			cout << (d.dis1[s][t] > INT_MAX ? -1 : d.dis1[s][t]) << endl;
		}
		return 0;
	}
	for (int i = 0; i <= 22; ++i) dij2(i);
	while (q--) {
		int s, t; cin >> s >> t; int ans = LLONG_MAX;
		if (s == t) {
			cout << 2 * a[s] << endl;
			continue;
		}
		for (int i = 0; i <= 22; ++i)
			if (a[s] & (1ll << i))
				ans = min(ans, a[s] + d.dis2[i][t]);
		cout << (ans > INT_MAX ? -1 : ans) << endl;
	}
	return 0;
}
