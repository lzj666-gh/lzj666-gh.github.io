#include<bits/stdc++.h>
#define all(a) (a).begin(), (a).end()
#define rall(a) (a).rbegin(), (a).rend()
#define int long long 
using namespace std;
int dp[100000 + 5][3], a[100000 + 5], n;
char opt[100000 + 5];
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> a[1]; opt[1] = '+';
	for (int i = 2; i <= n; ++i) cin >> opt[i] >> a[i];
	memset(dp, 0xcf, sizeof dp);
	dp[0][0] = 0;
	for (int i = 1; i <= n; ++i) {
		if (opt[i] == '+') {
			dp[i][0] = max(dp[i][0], dp[i - 1][0] + a[i]);
			dp[i][1] = max(dp[i][1], dp[i - 1][1] - a[i]);
			dp[i][2] = max(dp[i][2], dp[i - 1][2] + a[i]);
			dp[i][0] = max(dp[i][0], dp[i - 1][1] - a[i]);
			dp[i][1] = max(dp[i][1], dp[i - 1][2] + a[i]);
		}
		else {
			dp[i][0] = max(dp[i][0], dp[i - 1][0] - a[i]);
			dp[i][1] = max(dp[i][1], dp[i - 1][1] + a[i]);
			dp[i][2] = max(dp[i][2], dp[i - 1][2] - a[i]);
			dp[i][0] = max(dp[i][0], dp[i - 1][1] + a[i]);
			dp[i][1] = max(dp[i][1], dp[i - 1][2] - a[i]);
			dp[i][1] = max(dp[i][1], dp[i - 1][0] - a[i]);
			dp[i][2] = max(dp[i][2], dp[i - 1][1] + a[i]);
		}
	}
	cout << dp[n][0] << endl;
	return 0;
}
