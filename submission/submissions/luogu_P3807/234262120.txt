using namespace std;
int qpow(int a, int b, int mod) {
	int res = 1;
	while (b) {
		if (b & 1) res = res * a % mod;
		a = a * a % mod; b >>= 1;
	}
	return res;
}
template<int N = 100000000>
struct Linear_sieve {
	vector<int>prm;
	bool vis[N + 5];
	void prep() {
		for (int i = 2; i <= N; ++i) {
			if (!vis[i]) prm.push_back(i);
			for (auto& p : prm) {
				if (p * i > N) break;
				vis[p * i] = 1;
				if (i % p == 0) break;
			}
		}
	}
};
int fac[10000000 + 5], ifac[10000000 + 5];
int C(int n, int m, int p) {
	if (n < m) return 0;
	return fac[n] * ifac[m] % p * ifac[n - m] % p;
}
int Lucas(int n, int m, int p) {
	if (n < p && m < p) return C(n, m, p);
	return C(n % p, m % p, p) * Lucas(n / p, m / p, p) % p;
}
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	int T; cin >> T;
	while (T--) {
		int n, m, p; cin >> n >> m >> p; fac[0] = 1;
		for (int i = 1; i < p; ++i) fac[i] = fac[i - 1] * i % p; ifac[p - 1] = qpow(fac[p - 1], p - 2, p);
		for (int i = p - 2; i >= 0; --i) ifac[i] = ifac[i + 1] * (i + 1) % p;
		cout << Lucas(n + m, n, p) << endl;
	}
	return 0;
}
