#include<bits/stdc++.h>
#include<ext/pb_ds/assoc_container.hpp>
#include<ext/pb_ds/tree_policy.hpp>
#define int long long
#define file(name) freopen(#name".in", "r", stdin), freopen(#name".out", "w", stdout);
using namespace std;
using namespace __gnu_pbds;
tree<int, null_type, less<int>, rb_tree_tag, tree_order_statistics_node_update>t;
vector<int> ins[2000000 + 5];
int n, m, k, mx[2000000 + 5], my[2000000 + 5];
int32_t main() {
	//file(obstacle);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m >> k; int nn = n, mm = m;
	for (int i = 1; i <= n; ++i) mx[i] = m + 1;
	for (int i = 1; i <= m; ++i) my[i] = n + 1;
	for (int i = 1, x, y; i <= k; ++i)
		cin >> x >> y, mx[x] = min(mx[x], y), my[y] = min(my[y], x), ins[y].push_back(x),
		x == 1 && (m = min(m, y - 1)), y == 1 && (n = min(n, x - 1));
	long long ans = 0;
	for (int i = 1; i <= n; ++i) ans += mx[i] - 1;
	for (int i = n + 1; i <= nn; ++i) ins[1].push_back(i);
	for (int i = 1; i <= m; ++i) {
		ans += t.order_of_key(my[i]);
		for (auto& p : ins[i]) t.insert(p);
	}
	cout << ans << endl;
	return 0;
}
