#include<bits/stdc++.h>
#define int long long
using namespace std;
int exgcd(int a, int b, int& x, int& y) {
	if (b == 0) { x = 1; y = 0; return a; }
	int k = exgcd(b, a % b, y, x);
	y -= a / b * x;
	return k;
}
inline int inv(int a, int p) {
	int x, y; exgcd(a, p, x, y);
	return (x % p + p) % p;
}
int n, a[10], b[10];
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n;
	for (int i = 1; i <= n; ++i) cin >> a[i] >> b[i];
	int p = 1, ans = 0;
	for (int i = 1; i <= n; ++i) p *= a[i];
	for (int i = 1; i <= n; ++i) ans += b[i] * (p / a[i]) * inv(p / a[i], a[i]) % p, ans %= p;
	cout << ans % p << endl;
	return 0;
}
