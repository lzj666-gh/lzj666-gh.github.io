#include <bits/stdc++.h>
using namespace std;
const int MOD = 998244353;

int inv(int a) {
    int b = MOD, x0 = 0, x1 = 1;
    while (a > 1) {
        int q = a / b;
        int t = b;
        b = a % b;
        a = t;
        t = x0;
        x0 = x1 - q * x0;
        x1 = t;
    }
    if (x1 < 0) x1 += MOD;
    return x1;
}

int n, p, q;
int inv_p, inv_q;
long long total_prob = 0;
long long total_expect = 0;

void dfs(int minute, int k, long long prob, long long expect) {
    if (minute > n) {
        total_prob = (total_prob + prob) % MOD;
        total_expect = (total_expect + expect) % MOD;
        return;
    }
    
    // 摸鱼
    int fish_prob = (long long)(k + 1) * inv_p % MOD;
    dfs(minute + 1, 0, 
        prob * fish_prob % MOD, 
        (expect + prob * fish_prob % MOD * inv_q) % MOD);
    
    // 学习
    int study_prob = (1 - (long long)(k + 1) * inv_p % MOD + MOD) % MOD;
    dfs(minute + 1, k + 1, 
        prob * study_prob % MOD, 
        expect);
}

int main() {
    cin >> n >> p >> q;
    inv_p = inv(p);
    inv_q = inv(q);
    
    dfs(1, 0, 1, 0);
    
    cout << total_expect << endl;
    return 0;
}
