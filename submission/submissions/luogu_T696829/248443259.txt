#include<bits/stdc++.h>
#define int long long
using namespace std;
const int mod = 998244353;
int qpow(int a, int b) {
	int res = 1;
	while (b) {
		if (b & 1) res = res * a % mod;
		a = a * a % mod; b >>= 1;
	}
	return res;
}
vector<vector<int>> mul(const vector<vector<int>>& A, const vector<vector<int>>& B) {
	int n = A.size(), m = B[0].size(), p = B.size();
	vector<vector<int>> C(n, vector<int>(m, 0));
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < m; ++j)
			for (int k = 0; k < p; ++k)
				C[i][j] = (C[i][j] + A[i][k] * B[k][j] % mod) % mod;
	return C;
}
vector<vector<int>> qpow(vector<vector<int>> a, int b) {
	int n = a.size();
	vector<vector<int>> res(n, vector<int>(n, 0));
	for (int i = 0; i < n; ++i) res[i][i] = 1;
	while (b) {
		if (b & 1) res = mul(res, a);
		a = mul(a, a); b >>= 1;
	}
	return res;
}
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	int n, p, q; cin >> n >> p >> q;
	int inv_p = qpow(p, mod - 2), inv_q = qpow(q, mod - 2);
	vector<vector<int>> M(p + 1, vector<int>(p + 1, 0));
	for (int k = 0; k < p; k++) {
		int p1 = (k + 1) * inv_p % mod, p2 = (1 - p1 + mod) % mod;
		if(k == p - 1) p1 = 1, p2 = 0;
		M[0][k] = (M[0][k] + p1) % mod;
		M[p][k] = (M[p][k] + p1) % mod;
		if (p2 > 0 && k < p - 1)
			M[k + 1][k] = (M[k + 1][k] + p2) % mod;
	}
	M[p][p] = 1;
	vector<vector<int>> res(p + 1, vector<int>(1, 0)); res[0][0] = 1;
	res = mul(qpow(M, n), res); cout << res[p][0] * inv_q % mod << endl;
	return 0;
}
