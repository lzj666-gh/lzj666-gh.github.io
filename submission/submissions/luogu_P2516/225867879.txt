#include<bits/stdc++.h>
using namespace std;
constexpr int mod = 1e8;
string s1, s2;
long long dp[2][5010], len[2][5010];
signed main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> s1 >> s2;
	for (int j = 0; j <= s2.size(); ++j) len[0][j] = 1;
	for (int i = 1; i <= s1.size(); ++i) {
		memset(dp[i & 1], 0, sizeof dp[i & 1]);
		for (int j = 0; j <= s2.size(); ++j) len[i & 1][j] = 1;
		for (int j = 1; j <= s2.size(); ++j) {
			if (dp[i & 1][j] < dp[i & 1][j - 1]) dp[i & 1][j] = dp[i & 1][j - 1], len[i & 1][j] = len[i & 1][j - 1];
			else if (dp[i & 1][j] == dp[i & 1][j - 1]) len[i & 1][j] += len[i & 1][j - 1];
			if (dp[i & 1][j] < dp[!(i & 1)][j]) dp[i & 1][j] = dp[!(i & 1)][j], len[i & 1][j] = len[!(i & 1)][j];
			else if (dp[i & 1][j] == dp[!(i & 1)][j]) len[i & 1][j] += len[!(i & 1)][j];
			if (dp[i & 1][j] < dp[!(i & 1)][j - 1] + (s1[i - 1] == s2[j - 1])) dp[i & 1][j] = dp[!(i & 1)][j - 1] + (s1[i - 1] == s2[j - 1]), len[i & 1][j] = len[!(i & 1)][j - 1];
			else if (dp[i & 1][j] == dp[!(i & 1)][j - 1] + (s1[i - 1] == s2[j - 1])) len[i & 1][j] += len[!(i & 1)][j - 1];
			len[i & 1][j] %= mod;
		}
	}
	cout << dp[s1.size() & 1][s2.size()] - 1 << endl << len[s1.size() & 1][s2.size()] << endl;
	return 0;
}
