#include<bits/stdc++.h>
#define all(a) (a).begin(), (a).end()
#define rall(a) (a).rbegin(), (a).rend()
#define int long long
using namespace std;
constexpr int mod = 100003;
int qpow(int a, int b) {
	int res = 1;
	while (b) {
		if (b & 1) res = res * a % mod;
		a = a * a % mod; b >>= 1;
	}
	return res;
}
vector<int>prm;
bool vis[100000000 + 5];
void prep() {
	for (int i = 2; i <= 1e8; ++i) {
		if (!vis[i]) prm.push_back(i);
		for (auto& p : prm) {
			if (p * i > 1e8) break;
			vis[p * i] = 1;
			if (i % p == 0) break;
		}
	}
}
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	prep(); int n, q;
	cin >> n >> q;
	while (q--) {
		int k; cin >> k;
		cout << prm[k - 1] << endl;
	}
	return 0;
}
