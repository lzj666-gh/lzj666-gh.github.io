#include<bits/stdc++.h>
#define int long long
using namespace std;
struct Edge {
	int v, w, nxt;
}edge[200000 + 5];
int head[100000 + 5], idx;
void add(int u, int v, int w) {
	edge[++idx] = { v,w,head[u] };
	head[u] = idx;
}
int n, m, h[3000 + 5], dis[3000 + 5];
bool vis[3000 + 5];
void dij(int s) {
	memset(dis + 1, 0x3f, sizeof(int) * n);
	memset(vis + 1, 0, sizeof(bool) * n);
	priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>>q;
	dis[s] = 0; q.emplace(0, s);
	while (!q.empty()) {
		int now = q.top().second; q.pop();
		if (vis[now]) continue; vis[now] = 1;
		for (int i = head[now]; i; i = edge[i].nxt)
			if (int w = edge[i].w + dis[now]; w < dis[edge[i].v])
				dis[edge[i].v] = w, q.emplace(w, edge[i].v);
	}
}
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m;
	for (int i = 1, u, v, w; i <= m; ++i)
		cin >> u >> v >> w, add(u, v, w);
	for (int i = 1; i <= n; ++i) add(0, i, 0);
	memset(h, 0x3f, sizeof h); h[0] = 0;
	for (int k = 1; k <= n + 1; ++k)
		for (int i = 0; i <= n; ++i)
			for (int j = head[i]; j; j = edge[j].nxt)
				h[edge[j].v] = min(h[i] + edge[j].w, h[edge[j].v]);
	bool flg = 0;
	for (int i = 1; i <= n; ++i)
		for (int j = head[i]; j; j = edge[j].nxt)
			flg |= h[i] + edge[j].w < h[edge[j].v];
	//for (int i = 1; i <= n; ++i) cerr << h[i] << " \n"[i == n];
	if (flg) {
		cout << -1 << endl;
		return 0;
	}
	for (int i = 1; i <= n; ++i)
		for (int j = head[i]; j; j = edge[j].nxt)
			edge[j].w += h[i] - h[edge[j].v];
	for (int i = 1; i <= n; ++i) {
		long long ans = 0; dij(i);
		for (int j = 1; j <= n; ++j)
			if (vis[j]) ans += (dis[j] - (h[i] - h[j])) * j;
			else ans += 1e9 * j;
		cout << ans << endl;
	}
	return 0;
}
