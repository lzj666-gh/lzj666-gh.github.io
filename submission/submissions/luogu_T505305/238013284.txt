int head[200000 + 5], idx;
void add(int u, int v) {
	edge[++idx] = { v,head[u]};
	head[u] = idx;
}
int siz[200000 + 5];
void dfs(int x, int fa) {
	siz[x] = c[x];
	for (int i = head[x]; i; i = edge[i].nxt)
		if (edge[i].v ^ fa)
			dfs(edge[i].v, x), siz[x] += siz[edge[i].v];
}
