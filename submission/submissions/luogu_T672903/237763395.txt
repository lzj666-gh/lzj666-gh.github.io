#include<bits/stdc++.h>
#define all(a) (a).begin(), (a).end()
#define rall(a) (a).rbegin(), a.rend()
#define file(a) freopen(#a".in", "r", stdin), freopen(#a".out", "w", stdout)
using namespace std;
constexpr int N = 3e7 + 5;
vector<int>prm;
bool vis[N];
int g[N], f[N];
vector<int>v[1000];
void prep() {
	f[1] = 1; g[1] = 1;
	for (int i = 2; i < N; ++i) {
		if (!vis[i]) prm.push_back(i), f[i] = 2, g[i] = 1;
		for (int& k : prm) {
			if (k * i >= N) break;
			vis[k * i] = 1;
			if (i % k == 0) {
				f[i * k] = f[i] / (g[i] + 1) * (g[i] + 2);
				g[i * k] = g[i] + 1;
				break;
			}
			else {
				f[i * k] = f[i] * f[k];
				g[i * k] = 1;
			}
		}
	}
	for (int i = 1; i < N; ++i) v[f[i]].push_back(i);
}
int main() {
	// file(beats);
	cin.tie(0)->sync_with_stdio(false);
	prep(); int q, R; cin >> q >> R;
	while (q--) {
		int l, r, k; cin >> l >> r >> k;
		if (k < 0 || k > 600) { cout << 0 << endl; continue; }
		cout << (upper_bound(all(v[k]), r) - lower_bound(all(v[k]), l)) << endl;
	}
	return 0;
}
