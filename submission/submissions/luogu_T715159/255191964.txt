#include<bits/stdc++.h>
using namespace std;
int main() {
	cin.tie(0)->sync_with_stdio(false);
	int n; cin >> n;
	vector<int>a(n), p(n);
	for (int i = 0; i < n; ++i) cin >> a[i] >> p[i];
	for (int i = 0; i < n; ++i) cin;
	long long cnt = 0, ans = 0;
	for (int i = 0; i < n; ++i) {
		if (a[i] >= 500 && p[i] < 500) {
			cnt++; ans += p[i];
		}
	}
	cout << cnt << ' ' << ans << endl;
	return 0;
}
