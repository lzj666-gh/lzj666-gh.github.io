using namespace std;
constexpr unsigned int mod = 998244353;
unsigned int fac[50000 + 5], ifac[50000 + 5], n, m, k, a[50000 + 5];
struct Seg_Tree {
#define lc(p) (p << 1)
#define rc(p) (p << 1 | 1)
#define mid (l + r >> 1)
#define lson lc(p), l, mid
#define rson rc(p), mid + 1, r
#define self p, l, r
	struct Node {
		unsigned int cnt[10], ans, sum;
		bool lzy[10];
		Node() :cnt(), ans(), sum(), lzy() {}
		void cleartag() { fill(lzy, lzy + 10, 0); }
		void clearcnt() { fill(cnt, cnt + 10, 0); }
		friend Node operator+(const Node& a, const Node& b) {
			Node res;
			for (unsigned int i = 0; i < k; ++i)
				res.cnt[i] = a.cnt[i] + b.cnt[i];
			res.ans = (a.ans + b.ans) % mod;
			res.sum = (a.sum + b.sum) % mod;
			return res;
		}
	}tree[50000 << 2 | 3];
	void tag(unsigned int p, unsigned int l, unsigned int r, unsigned int i) {
		tree[p].clearcnt(); tree[p].cleartag(); tree[p].lzy[i] = 1;
		tree[p].ans = tree[p].sum * i % mod; tree[p].cnt[i] = r - l + 1;
	}
	void pushup(unsigned int p) {
		tree[p] = tree[lc(p)] + tree[rc(p)];
	}
	void pushdown(unsigned int p, unsigned int l, unsigned int r) {
		for (unsigned int i = 0; i < k; ++i)
			if (tree[p].lzy[i]) {
				tag(lson, i);
				tag(rson, i);
				tree[p].lzy[i] = 0;
				return;
			}
	}
	void build(unsigned int p, unsigned int l, unsigned int r) {
		if (l == r) {
			tree[p].cnt[a[l]] = 1;
			tree[p].ans = fac[l] * a[l] % mod;
			tree[p].sum = fac[l];
			return;
		}
		build(lson);
		build(rson);
		pushup(p);
	}
	void set(unsigned int p, unsigned int l, unsigned int r, unsigned int x, unsigned int k) {
		if (l == r) {
			tree[p].clearcnt();
			tree[p].cnt[k] = 1;
			tree[p].ans = fac[l] * k % mod;
			return;
		}
		pushdown(self);
		if (x <= mid) set(lson, x, k);
		else set(rson, x, k);
		pushup(p);
	}
	void set(unsigned int p, unsigned int l, unsigned int r, unsigned int x, unsigned int y, unsigned int k) {
		if (x <= l && r <= y) return tag(self, k);
		pushdown(self);
		if (x <= mid) set(lson, x, y, k);
		if (mid < y) set(rson, x, y, k);
		pushup(p);
	}
	Node query(unsigned int p, unsigned int l, unsigned int r, unsigned int x, unsigned int y) {
		if (x <= l && r <= y) return tree[p];
		if (y < l || r < x) return Node();
		pushdown(self);
		return query(lson, x, y) + query(rson, x, y);
	}
}t;
unsigned int qpow(unsigned int a, unsigned int b) {
	unsigned int res = 1;
	while (b) {
		if (b & 1) res = res * a % mod;
		a = a * a % mod; b >>= 1;
	}
	return res;
}

	cin >> n >> m >> k; string s; cin >> s;
	for (unsigned int i = 1; i <= n; ++i) a[i] = s[i - 1] - '0';
	for (unsigned int i = 1; i <= n; ++i) fac[i] = qpow(k, n - i + 1);
	for (unsigned int i = 1; i <= n; ++i) ifac[i] = qpow(fac[i], mod - 2);
	t.build(1, 1, n);
	while (m--) {
		unsigned int opt, x, y; cin >> opt >> x >> y;
		if (opt == 1) {
			t.set(1, 1, n, x, y);
		}
		if (opt == 2) {
			auto p = t.query(1, 1, n, x, y);
			for (unsigned int i = 0; i < k; ++i)
				t.set(1, 1, n, x, x + p.cnt[i] - 1, i), x += p.cnt[i];
		}
		if (opt == 3) {
			auto p = t.query(1, 1, n, x, y);
			for (unsigned int i = k - 1; i >= 0; --i)
				t.set(1, 1, n, x, x + p.cnt[i] - 1, i), x += p.cnt[i];
		}
		if (opt == 4) {
