#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
#include <numeric>
#include <cmath>
#include <map>
#include <bits/stdc++.h>
using namespace std;

int gcd(int a, int b) {
    if (b == 0) return a;
    return gcd(b, a % b);
}

set<vector<int>> solutions;

void solve(int k, int a1, const vector<int>& factors) {
    vector<int> g(k);
    set<vector<int>> sols;

    function<void(int)> dfs_g = [&](int idx) {
        if (idx == k) {
            if (a1 % g[k-1] != 0) return;
            int k1 = a1 / g[k-1];
            int L1 = (g[0] + g[k-1] - 1) / g[k-1];
            if (k1 < L1) return;

            int S = accumulate(g.begin(), g.end(), 0);
            vector<int> L;
            for (int i = 1; i < k; i++) {
                L.push_back((g[i] + g[i-1] - 1) / g[i-1]);
            }

            vector<int> k_vals(k-1);
            function<void(int, int)> dfs_k = [&](int i, int cur) {
                if (i == k-1) {
                    if (cur == S) {
                        vector<int> a = {a1};
                        for (int j = 0; j < k-1; j++) {
                            a.push_back(k_vals[j] * g[j]);
                        }
                        bool valid = true;
                        for (int i0 = 0; i0 < k; i0++) {
                            int j = (i0+1) % k;
                            int g_val = gcd(a[i0], a[j]);
                            if (g_val != g[i0]) {
                                valid = false;
                                break;
                            }
                        }
                        if (valid) {
                            sols.insert(a);
                        }
                    }
                    return;
                }
                int step = g[i];
                int start = L[i];
                int remaining = S - cur;
                int max_val = remaining / step;
                for (int v = start; v <= max_val; v++) {
                    k_vals[i] = v;
                    dfs_k(i+1, cur + step * v);
                }
            };
            dfs_k(0, 0);
            return;
        }
        for (int d : factors) {
            g[idx] = d;
            dfs_g(idx+1);
        }
    };

    dfs_g(0);
    solutions.insert(sols.begin(), sols.end());
}

int main() {
    int k, a1;
    cin >> k >> a1;
    if (k == 1) {
        cout << 0 << endl;
        return 0;
    }

    vector<int> factors;
    for (int i = 1; i * i <= a1; i++) {
        if (a1 % i == 0) {
            factors.push_back(i);
            if (i != a1 / i) {
                factors.push_back(a1 / i);
            }
        }
    }
    sort(factors.begin(), factors.end());
    factors.erase(unique(factors.begin(), factors.end()), factors.end());

    solve(k, a1, factors);

    cout << solutions.size() << endl;
    for (const auto& sol : solutions) {
        for (int i = 0; i < sol.size(); i++) {
            if (i > 0) cout << " ";
            cout << sol[i];
        }
        cout << endl;
    }

    return 0;
}
