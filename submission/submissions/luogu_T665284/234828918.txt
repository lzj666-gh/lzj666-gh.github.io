#include<bits/stdc++.h>
using namespace std;
#define int long long
int n, m, t, c; vector<int>p, s, v;
vector<vector<int>>ans;
signed main() {
	ios::sync_with_stdio(0); cin >> n >> m;
	for (int i = 0; i != n; ++i) s.emplace_back(m);
	for (int i = 1; i * i <= m; ++i)
		if (m % i == 0)
			for (int j = 1; j != n - 1; ++j)
				for (int k = j + 2; k != n; ++k) {
					p = s;
					for (int l = j; l <= k; ++l) p[l] = i;
					for (int l = j + 1; l != k; ++l) {
						v = p;
						for (int r = l; r != k; ++r)
							v[r] = 2 * i,
							ans.emplace_back(v);
					}
				}
	cout << ans.size() << endl;
	for (const auto& t : ans) {
		for (int v : t) cout << v << " ";
		cout << endl;
	}
}
