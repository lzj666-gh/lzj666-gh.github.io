#include<bits/stdc++.h>
using namespace std;
#define int long long
int n,a0;
signed main(){
	cin>>n>>a0;
	cout<<(n-1)*(n-2)/2<<"\n";
	for(int i = 1;i<n-1;i++){
		for(int j = i+1;j<n;j++){
			for(int k = 0;k<i;k++){
				cout<<a0<<" ";
			}
			for(int k = i;k<=j;k++){
				cout<<2*a0<<" ";
			}
			for(int k = j+1;k<n;k++){
				cout<<a0<<" ";
			}
			cout<<"\n";
		}
	}
}
