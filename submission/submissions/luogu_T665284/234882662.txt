#define _USE_MATH_DEFINES
#include<bits/stdc++.h>
#define int long long
using namespace std;
int k, a1; vector<vector<int>>ans;
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> k >> a1;
	vector<int>v(k + 1, a1);
	for (int d = 1; d <= a1; ++d) {
		if (a1 % d == 0) {
			for (int l1 = 2; l1 <= k; ++l1) {
				for (int r1 = l1; r1 <= k; ++r1) {
					for (int i = l1; i <= r1; ++i) v[i] = d;
					for (int l2 = l1 + 1; l2 < r1; ++l2)
						for (int r2 = l2; r2 < r1; ++r2) {
							for (int i = l2; i <= r2; ++i) v[i] = d * 2;
							ans.push_back(v);
							for (int i = l2; i <= r2; ++i) v[i] = d;
						}
					for (int i = l1; i <= r1; ++i) v[i] = a1;
				}
			}
		}
	}
	cout << ans.size() << endl;
	for (auto& k : ans)
		for (int i = 1; i < k.size(); ++i)
			cout << k[i] << " \n"[i == k.size() - 1];
	return 0;
}
