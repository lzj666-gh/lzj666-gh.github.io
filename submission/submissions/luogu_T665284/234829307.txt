#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
using namespace std;

vector<int> get_divisors(int n) {
    vector<int> divisors;
    for (int i = 1; i * i <= n; i++) {
        if (n % i == 0) {
            divisors.push_back(i);
            if (i != n / i) {
                divisors.push_back(n / i);
            }
        }
    }
    sort(divisors.begin(), divisors.end());
    return divisors;
}

int main() {
    int k, a1;
    cin >> k >> a1;
    if (k == 1) {
        cout << 0 << endl;
        return 0;
    }
    vector<int> divisors = get_divisors(a1);
    vector<vector<int>> solutions;
    for (int g : divisors) {
        bool multiple_of_2g = false;
        if (2 * g <= a1) {
            if (a1 % (2 * g) == 0) {
                multiple_of_2g = true;
            }
        }
        if (!multiple_of_2g) {
            for (int j = 2; j <= k; j++) {
                vector<int> sol;
                sol.push_back(a1);
                for (int i = 2; i <= k; i++) {
                    if (i == j) {
                        sol.push_back(2 * g);
                    } else {
                        sol.push_back(g);
                    }
                }
                solutions.push_back(sol);
            }
        } else {
            for (int j = 3; j < k; j++) {
                vector<int> sol;
                sol.push_back(a1);
                for (int i = 2; i <= k; i++) {
                    if (i == j) {
                        sol.push_back(2 * g);
                    } else {
                        sol.push_back(g);
                    }
                }
                solutions.push_back(sol);
            }
        }
    }
    cout << solutions.size() << endl;
    for (auto sol : solutions) {
        for (size_t i = 0; i < sol.size(); i++) {
            if (i > 0) cout << " ";
            cout << sol[i];
        }
        cout << endl;
    }
    return 0;
}
