#include<bits/stdc++.h>
using namespace std;
int n, m, a[2000000 + 5];
struct Line {
	int idx, x, dst, coef;
	bool operator<(const Line& oth) { return idx < oth.idx; }
}line[4000000 + 5];
int ans[2000000 + 5];
struct FWT {
	int s[2000000 + 5];
	void add(int x, int k) { while (x <= 2e6) { s[x] += k; x += x & ~x + 1; } }
	int query(int x) { int res = 0; while (x) { res += s[x]; x -= x & ~x + 1; } return res; }
}t;
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m;
	for (int i = 1; i <= n; ++i)
		cin >> a[i];
	for (int i = 1; i <= m; ++i) {
		int l, r, x; cin >> l >> r >> x;
		line[i - 1 << 1] = { l - 1,x,i,-1 };
		line[i - 1 << 1 | 1] = { r,x,i,1 };
	}
	sort(line, line + (m <<= 1));
	for (int i = 0, j = 0; i < m; ++i) {
		while (j < line[i].idx) t.add(a[++j], 1);
		ans[line[i].dst] += line[i].coef * t.query(line[i].x);
	}
	m >>= 1;
	for (int i = 1; i <= m; ++i) cout << ans[i] << '\n';
	cout << endl;
	return 0;
}
