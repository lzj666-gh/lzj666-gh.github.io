#include<bits/stdc++.h>
#define 删除(翁法罗斯) memset(翁法罗斯, 0, sizeof 翁法罗斯)
using namespace std;
int 长不高的, 会说话的, 诗[200000 + 5], 随风逝去后[200000 + 5], 仍被留下的[200000 + 5];
struct 昔涟 {
	int l, r, idx;
	friend bool operator<(const 昔涟& 桃子, const 昔涟& 爱) {
		return 诗[桃子.l] == 诗[爱.l] ? 桃子.r < 爱.r : 桃子.l < 爱.l;
	}
}浪漫故事[200000 + 5];
int 起始[200000 + 5], 终结[200000 + 5], 哭着诞生的[200000 + 5], 笑着道别的[200000 + 5], 粉色[500];
bool 收梢[200000 + 5], philia093[200000 + 5]; int 花, 道别;
void 花开时(int 水晶) {
	if (收梢[随风逝去后[水晶]]) {
		起始[随风逝去后[水晶]] = min(起始[随风逝去后[水晶]], 水晶);
		终结[随风逝去后[水晶]] = max(终结[随风逝去后[水晶]], 水晶);
		花 = max(花, 终结[随风逝去后[水晶]] - 起始[随风逝去后[水晶]]);
	}
	else {
		起始[随风逝去后[水晶]] = 终结[随风逝去后[水晶]] = 水晶;
		收梢[随风逝去后[水晶]] = 1;
	}
}
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> 长不高的; vector<int>翁法罗斯;
	for (int i = 1; i <= 长不高的; ++i)
		cin >> 随风逝去后[i], 翁法罗斯.push_back(随风逝去后[i]);
	sort(翁法罗斯.begin(), 翁法罗斯.end());
	for (int i = 1; i <= 长不高的; ++i)
		随风逝去后[i] = lower_bound(翁法罗斯.begin(), 翁法罗斯.end(), 随风逝去后[i]) - 翁法罗斯.begin() + 1;
	cin >> 会说话的;
	for (int i = 1; i <= 会说话的; ++i) cin >> 浪漫故事[i].l >> 浪漫故事[i].r, 浪漫故事[i].idx = i;
	int 小尾巴 = sqrt(长不高的);
	for (int i = 1; i <= 长不高的; ++i) 诗[i] = (i - 1) / 小尾巴 + 1;
	for (int i = 1; i <= 诗[长不高的]; ++i) 粉色[i] = min(i * 小尾巴, 长不高的);
	sort(浪漫故事 + 1, 浪漫故事 + 会说话的 + 1);
	int 柔软的 = 0;
	for (int i = 1; i <= 会说话的; ++i) {
		if (诗[i] != 诗[i - 1]) {
			柔软的 = 粉色[诗[浪漫故事[i].l]]; 花 = 0;
			删除(起始); 删除(终结); 删除(收梢);
		}
		if (诗[浪漫故事[i].l] == 诗[浪漫故事[i].r]) {
			for (int j = 浪漫故事[i].l; j <= 浪漫故事[i].r; ++j) {
				if (!收梢[随风逝去后[j]])
					起始[随风逝去后[j]] = j,
					收梢[随风逝去后[j]] = 1;
				else
					花 = max(花, j - 起始[随风逝去后[j]]);
			}
			仍被留下的[浪漫故事[i].idx] = 花; 花 = 0;
			for (int j = 浪漫故事[i].l; j <= 浪漫故事[i].r; ++j)
				起始[随风逝去后[j]] = 收梢[随风逝去后[j]] = 0;
			continue;
		}
		while (柔软的 < 浪漫故事[i].r) 花开时(++柔软的);
		for (int j = 浪漫故事[i].l; j <= 粉色[诗[浪漫故事[i].l]]; ++j)
			哭着诞生的[随风逝去后[j]] = 起始[随风逝去后[j]],
			笑着道别的[随风逝去后[j]] = 终结[随风逝去后[j]],
			philia093[随风逝去后[j]] = 收梢[随风逝去后[j]];                                                                                         
		道别 = 花;
		for (int j = 浪漫故事[i].l; j <= 粉色[诗[浪漫故事[i].l]]; ++j)
			花开时(j);
		仍被留下的[浪漫故事[i].idx] = 花;
		for (int j = 浪漫故事[i].l; j <= 粉色[诗[浪漫故事[i].l]]; ++j)
			起始[随风逝去后[j]] = 哭着诞生的[随风逝去后[j]],
			终结[随风逝去后[j]] = 笑着道别的[随风逝去后[j]],
			收梢[随风逝去后[j]] = philia093[随风逝去后[j]];
		花 = 道别;
	}
	for (int 温柔 = 1; 温柔 <= 会说话的; ++温柔) cout << 仍被留下的[温柔] << endl;
	return 0;
}
