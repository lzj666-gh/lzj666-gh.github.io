#include<bits/stdc++.h>
#define file(name) freopen(#name".in", "r", stdin), freopen(#name".out", "w", stdout);
using namespace std;
using ll = long long;
ll v[200000 + 5], a[200000 + 5], sv[2000000 + 5], n, s, k, V, A;
double dp[200000 + 5];
bool check(double t) {
	double d1 = t, d2 = 1.0 / 2 * t * t;
	dp[0] = V * d1 + A * d2;
	for (int i = 1; i <= n; ++i) {
		dp[i] = dp[i - 1] + v[i] * d1;
		if (i >= k)
			dp[i] = max(dp[i], dp[i - k] + (sv[i - 1] - sv[i - k]) * d1 + a[i] * d2);
		else
			dp[i] = max(dp[i], sv[i - 1] * d1 + A * d2 + a[i] * d2);
	}
	return dp[n] >= s;
}
int32_t main() {
	// file(run);
	cin.tie(0)->sync_with_stdio(false);
	int T;
	cin >> T;
	while (T--) {
		cin >> n >> k >> s >> V >> A;
		sv[0] = V;
		for (int i = 1; i <= n; ++i) cin >> v[i], sv[i] = sv[i - 1] + v[i];
		for (int i = 1; i <= n; ++i) cin >> a[i];
		double l = -1, r = 1e7;
		if (A != 0) r = ceil((sqrt(V * V + 2 * A * s) - V) / A);
		while (l < r - 1e-10) {
			double mid = (l + r) / 2;
			if (check(mid)) r = mid;
			else l = mid;
		}
		cout << fixed << setprecision(10) << l << endl;
	}
	return 0;
}
