#include<bits/stdc++.h>
using namespace std;
int a[100000 + 5], b[100000 + 5], idx[100000 + 5], f[100000 + 5], n;
int main() {
	cin.tie(0)->sync_with_stdio(false);
	memset(f, 0x3f, sizeof f);
	cin >> n;
	for (int i = 1; i <= n; ++i) cin >> a[i], idx[a[i]] = i;
	for (int i = 1; i <= n; ++i) cin >> b[i], b[i] = idx[b[i]];
	int top = 0;
	f[0] = 0;
	for (int i = 1; i <= n; i++) {
		if (b[i] > f[top]) f[++top] = b[i];
		else {
			int k = lower_bound(f, f + top + 1, b[i]) - f;
			f[k] = min(b[i], f[k]);
		}
	}
	cout << top << endl;
	return 0;
}
