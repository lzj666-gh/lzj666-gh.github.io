#include<bits/stdc++.h>
#include<unordered_map>
#define all(a) (a).begin(), (a).end()
#define file(a) { freopen(#a".in", "r", stdin); freopen(#a".out", "w", stdout); }
using namespace std;
using ll = long long;
constexpr int mod = 998244353, N = 2000000, inv2 = (mod + 1) / 2;
ll qpow(ll a, ll b) {
	ll res = 1;
	while (b) {
		if (b & 1) res = res * a % mod;
		a = a * a % mod; b >>= 1;
	}
	return res;
}
bool vis[N];
ll phi[N], mu[N];
vector<int>prm;
void init() {
	phi[1] = mu[1] = 1;
	for (int i = 2; i < N; ++i) {
		if (!vis[i])
			prm.push_back(i), phi[i] = i - 1, mu[i] = -1;
		for (const int& p : prm) {
			if (i * p >= N) break;
			vis[i * p] = 1;
			if (i % p == 0) {
				phi[i * p] = phi[i] * p;
				mu[i * p] = 0;
				break;
			}
			phi[i * p] = phi[i] * (p - 1);
			mu[i * p] = -mu[i];
		}
	}
	for (int i = 1; i < N; ++i)
		phi[i] = (phi[i - 1] + phi[i]) % mod, mu[i] = (mu[i - 1] + mu[i]) % mod;
}
unordered_map<ll, ll> visphi;
ll sumphi(ll n) {
	if (n < N) return phi[n];
	if (visphi.count(n)) return visphi[n];
	ll& res = visphi[n]; res = n % mod * ((n + 1) % mod) % mod * inv2 % mod;
	for (ll l = 2, r; l <= n; l = r + 1) {
		r = n / (n / l);
		res = (res - (r - l + 1) * sumphi(n / l) % mod) % mod;
	}
	return res;
}
unordered_map<ll, ll> vismu;
ll summu(ll n) {
	if (n < N) return mu[n];
	if (vismu.count(n)) return vismu[n];
	ll& res = vismu[n]; res = 1;
	for (ll l = 2, r; l <= n; l = r + 1) {
		r = n / (n / l);
		res = (res - (r - l + 1) % mod * summu(n / l) % mod) % mod;
	}
	return res;
}
ll n, m;
unordered_map<ll, ll>inv;
int main() {
	// file(table);
	cin.tie(0)->sync_with_stdio(false);
	init(); cin >> n >> m;
	ll ans = 0; m %= mod - 1;
	for (ll l = 1, r; l <= n; l = r + 1) {
		r = n / (n / l); ll res = 0;
		for (ll l2 = 1, r2, lim = n / l; l2 <= lim; l2 = r2 + 1) {
			r2 = lim / (lim / l2);
			if (inv.count(n / (l * l2)))
				res = (res + (summu(r2) - summu(l2 - 1)) % mod * inv[n / (l * l2)] % mod) % mod;
			else res = (res + (summu(r2) - summu(l2 - 1)) % mod * (inv[n / (l * l2)] = qpow(n / (l * l2), m)) % mod) % mod;
		}
		ans = (ans + (r - l + 1) % mod * (2 * sumphi(n / l) - 1) % mod * res % mod) % mod;
	}
	cout << (ans + mod) % mod << endl;
	return 0;
}
