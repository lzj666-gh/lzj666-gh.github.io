#include<bits/stdc++.h>
using namespace std;
int n, m, k, x, y, p, res, ans, a[1005][1005];
unsigned state;
inline unsigned next_random() {
	state = state ^ (state << 13);
	state = state ^ (state >> 17);
	state = state ^ (state << 5);
	return state;
}
inline int random(int l, int r) {
	return next_random() % (r - l + 1) + l;
}
int sum[4][1005][1005];
signed main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m >> k >> x >> y >> p;
	for (int i = 0; i != n; ++i)
		for (int j = 0; j != m; ++j)
			cin >> a[i][j];
	cin >> state;
	for (int i = 0; i != n; ++i)
		for (int j = 0; j != m; ++j)
			sum[0][i][j] = sum[1][i][j] = sum[2][i][j] = sum[3][i][j] = a[i][j];
	for (int i = 0; i < n; i += x)
		for (int j = 0; j < m; j += y) {
			for (int k1 = 1; k1 < x && i + k1 < n; ++k1)
				for (int k2 = 0; k2 < y && j + k2 < m; ++k2)
					sum[0][i + k1][j + k2] = 1ll * sum[0][i + k1][j + k2] * sum[0][i + k1 - 1][j + k2] % p;
			for (int k1 = 0; k1 < x && i + k1 < n; ++k1)
				for (int k2 = 1; k2 < y && j + k2 < m; ++k2)
					sum[0][i + k1][j + k2] = 1ll * sum[0][i + k1][j + k2] * sum[0][i + k1][j + k2 - 1] % p;
		}
	for (int i = 0; i < n; i += x)
		for (int j = 0; j < m; j += y) {
			int j_end = min(j + y - 1, m - 1);
			for (int k1 = 0; k1 < x && i + k1 < n; ++k1)
				for (int k2 = j_end - 1; k2 >= j; --k2)
					sum[1][i + k1][k2] = 1ll * sum[1][i + k1][k2] * sum[1][i + k1][k2 + 1] % p;
			for (int k2 = j; k2 <= j_end; ++k2)
				for (int k1 = 1; k1 < x && i + k1 < n; ++k1)
					sum[1][i + k1][k2] = 1ll * sum[1][i + k1][k2] * sum[1][i + k1 - 1][k2] % p;
		}
	for (int i = 0; i < n; i += x)
		for (int j = 0; j < m; j += y) {
			int i_end = min(i + x - 1, n - 1);
			for (int k1 = 0; k1 < x && i + k1 < n; ++k1)
				for (int k2 = j + 1; k2 < j + y && k2 < m; ++k2)
					sum[2][i + k1][k2] = 1ll * sum[2][i + k1][k2] * sum[2][i + k1][k2 - 1] % p;
			for (int k2 = j; k2 < j + y && k2 < m; ++k2)
				for (int k1 = i_end - 1; k1 >= i; --k1)
					sum[2][k1][k2] = 1ll * sum[2][k1][k2] * sum[2][k1 + 1][k2] % p;
		}
	for (int i = 0; i < n; i += x)
		for (int j = 0; j < m; j += y) {
			int i_end = min(i + x - 1, n - 1), j_end = min(j + y - 1, m - 1);
			for (int k1 = 0; k1 < x && i + k1 < n; ++k1)
				for (int k2 = j_end - 1; k2 >= j; --k2)
					sum[3][i + k1][k2] = 1ll * sum[3][i + k1][k2] * sum[3][i + k1][k2 + 1] % p;
			for (int k2 = j; k2 <= j_end; ++k2)
				for (int k1 = i_end - 1; k1 >= i; --k1)
					sum[3][k1][k2] = 1ll * sum[3][k1][k2] * sum[3][k1 + 1][k2] % p;
		}
	//for (int i = 0; i < n; ++i, cerr << endl)
	//	for (int j = 0; j < m; ++j)
	//		cerr << sum[0][i][j] << ' '; cerr << endl;
	//for (int i = 0; i < n; ++i, cerr << endl)
	//	for (int j = 0; j < m; ++j)
	//		cerr << sum[1][i][j] << ' '; cerr << endl;
	//for (int i = 0; i < n; ++i, cerr << endl)
	//	for (int j = 0; j < m; ++j)
	//		cerr << sum[2][i][j] << ' '; cerr << endl;
	//for (int i = 0; i < n; ++i, cerr << endl)
	//	for (int j = 0; j < m; ++j)
	//		cerr << sum[3][i][j] << ' '; cerr << endl;
	int min_x = x + 1, max_x = min(x * 2, n), min_y = y + 1, max_y = min(y * 2, m);
	while (k--) {
		int len_x = random(min_x, max_x), len_y = random(min_y, max_y);
		int xmin = random(0, n - len_x), ymin = random(0, m - len_y);
		int xmax = xmin + len_x - 1, ymax = ymin + len_y - 1;
		//cerr << xmin << ' ' << xmax << ' ' << ymin << ' ' << ymax << endl;
		ans = 1ll * sum[3][xmin][ymin] * sum[1][xmax][ymin] % p * sum[2][xmin][ymax] % p * sum[0][xmax][ymax] % p;
		if (xmin / x == xmax / x - 1)
			if (ymin / y == ymax / y - 1)
				ans = 1ll * ans * 1 % p;
			else
				ans = 1ll * ans * sum[2][xmin][ymax / y * y - 1] % p * sum[0][xmax][ymax / y * y - 1] % p;
		else
			if (ymin / y == ymax / y - 1)
				ans = 1ll * ans * sum[1][xmax / x * x - 1][ymin] % p * sum[0][xmax / x * x - 1][ymax] % p;
			else
				ans = 1ll * ans * sum[1][xmax / x * x - 1][ymin] % p * sum[0][xmax / x * x - 1][ymax] % p,
				ans = 1ll * ans * sum[2][xmin][ymax / y * y - 1] % p * sum[0][xmax][ymax / y * y - 1] % p,
				ans = 1ll * ans * sum[0][xmax / x * x - 1][ymax / y * y - 1] % p;
		res ^= ans;
	}
	cout << res << endl;
	return 0;
}
