#include<bits/stdc++.h>
using namespace std;
struct IO {
#define mxsz (1 << 21)
	char buf[mxsz], * p1, * p2;
	char pbuf[mxsz], * pp; bool acc[128];
	IO() : p1(buf), p2(buf), pp(pbuf) {}
	~IO() { fwrite(pbuf, 1, pp - pbuf, stdout); }
	inline char gc() {
		if (p1 == p2) p2 = (p1 = buf) + fread(buf, 1, mxsz, stdin);
		return p1 == p2 ? ' ' : *p1++;
	}
	inline void setacc(const char* c) { while (*c) acc[*c++] = 1; }
	inline char getc() { char c; while (!acc[c = gc()]); return c; }
	inline int read() {
		int r = 0; char c = gc(); while (c < '0' || c>'9') c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return r;
	}
	inline void push(const char& c) {
		if (pp - pbuf == mxsz) fwrite(pbuf, 1, mxsz, stdout), pp = pbuf;
		*pp++ = c;
	}
	inline void write(int x) {
		static char sta[22]; int top = 0;
		do sta[top++] = x % 10, x /= 10; while (x);
		while (top) push(sta[--top] ^ 48);
	}
	inline void write(int x, char opc) {
		write(x), push(opc);
	}
	inline void puts(const char* c) { while (*c) push(*c++); }
} io;
int n, m, k, x, y, p, res, ans, a[1005][1005];
unsigned state;
inline unsigned next_random() {
	state = state ^ (state << 13);
	state = state ^ (state >> 17);
	state = state ^ (state << 5);
	return state;
}
inline int random(int l, int r) {
	return next_random() % (r - l + 1) + l;
}
int sum[4][1005][1005];
signed main() {
	ios::sync_with_stdio(0);
	n = io.read(); m = io.read(); k = io.read();
	x = io.read(); y = io.read(); p = io.read();
	for (int i = 0; i != n; ++i)
		for (int j = 0; j != m; ++j)
			a[i][j] = io.read();
	state = io.read();
	for (int i = 0; i != n; ++i)
		for (int j = 0; j != m; ++j)
			sum[0][i][j] = sum[1][i][j] = sum[2][i][j] = sum[3][i][j] = a[i][j];
	for (int i = 0; i < n; i += x)
		for (int j = 0; j < m; j += y) {
			for (int k1 = 1; k1 < x && i + k1 < n; ++k1)
				for (int k2 = 0; k2 < y && j + k2 < m; ++k2)
					sum[0][i + k1][j + k2] = 1ll * sum[0][i + k1][j + k2] * sum[0][i + k1 - 1][j + k2] % p;
			for (int k1 = 0; k1 < x && i + k1 < n; ++k1)
				for (int k2 = 1; k2 < y && j + k2 < m; ++k2)
					sum[0][i + k1][j + k2] = 1ll * sum[0][i + k1][j + k2] * sum[0][i + k1][j + k2 - 1] % p;
		}
	for (int i = 0; i < n; i += x)
		for (int j = 0; j < m; j += y) {
			int j_end = min(j + y - 1, m - 1);
			for (int k1 = 0; k1 < x && i + k1 < n; ++k1)
				for (int k2 = j_end - 1; k2 >= j; --k2)
					sum[1][i + k1][k2] = 1ll * sum[1][i + k1][k2] * sum[1][i + k1][k2 + 1] % p;
			for (int k2 = j; k2 <= j_end; ++k2)
				for (int k1 = 1; k1 < x && i + k1 < n; ++k1)
					sum[1][i + k1][k2] = 1ll * sum[1][i + k1][k2] * sum[1][i + k1 - 1][k2] % p;
		}
	for (int i = 0; i < n; i += x)
		for (int j = 0; j < m; j += y) {
			int i_end = min(i + x - 1, n - 1);
			for (int k1 = 0; k1 < x && i + k1 < n; ++k1)
				for (int k2 = j + 1; k2 < j + y && k2 < m; ++k2)
					sum[2][i + k1][k2] = 1ll * sum[2][i + k1][k2] * sum[2][i + k1][k2 - 1] % p;
			for (int k2 = j; k2 < j + y && k2 < m; ++k2)
				for (int k1 = i_end - 1; k1 >= i; --k1)
					sum[2][k1][k2] = 1ll * sum[2][k1][k2] * sum[2][k1 + 1][k2] % p;
		}
	for (int i = 0; i < n; i += x)
		for (int j = 0; j < m; j += y) {
			int i_end = min(i + x - 1, n - 1), j_end = min(j + y - 1, m - 1);
			for (int k1 = 0; k1 < x && i + k1 < n; ++k1)
				for (int k2 = j_end - 1; k2 >= j; --k2)
					sum[3][i + k1][k2] = 1ll * sum[3][i + k1][k2] * sum[3][i + k1][k2 + 1] % p;
			for (int k2 = j; k2 <= j_end; ++k2)
				for (int k1 = i_end - 1; k1 >= i; --k1)
					sum[3][k1][k2] = 1ll * sum[3][k1][k2] * sum[3][k1 + 1][k2] % p;
		}
//	for (int i = 0; i < n; ++i, cerr << endl)
//		for (int j = 0; j < m; ++j)
//			cerr << sum[0][i][j] << ' '; cerr << endl;
//	for (int i = 0; i < n; ++i, cerr << endl)
//		for (int j = 0; j < m; ++j)
//			cerr << sum[1][i][j] << ' '; cerr << endl;
//	for (int i = 0; i < n; ++i, cerr << endl)
//		for (int j = 0; j < m; ++j)
//			cerr << sum[2][i][j] << ' '; cerr << endl;
//	for (int i = 0; i < n; ++i, cerr << endl)
//		for (int j = 0; j < m; ++j)
//			cerr << sum[3][i][j] << ' '; cerr << endl;
	int mic = x + 1, mac = min(x * 2, n), mir = y + 1, mar = min(y * 2, m);
	while (k--) {
		int lc = random(mic, mac), lr = random(mir, mar);
		int xl = random(0, n - lc), yl = random(0, m - lr);
		int xr = xl + lc - 1, yr = yl + lr - 1;
//		cerr << xl << ' ' << xr << ' ' << yl << ' ' << yr << endl;
		ans = 1ll * sum[3][xl][yl] * sum[1][xr][yl] % p * sum[2][xl][yr] % p * sum[0][xr][yr] % p;
//		cerr << ans << endl;
		if (xl / x == xr / x - 1)
			if (yl / y == yr / y - 1)
				ans = ans * 1 % p;
			else
				ans = ans * sum[2][xr / x * x - 1][yl] % p * sum[3][xr / x * x - 1][yr] % p;
		else
			if (yl / y == yr / y - 1)
				ans = ans * sum[1][xl][yr / y * y - 1] % p * sum[3][xr][yr / y * y - 1] % p;
			else
				ans = ans * sum[2][xr / x * x - 1][yl] % p * sum[3][xr / x * x - 1][yr] % p,
				ans = ans * sum[1][xl][yr / y * y - 1] % p * sum[3][xr][yr / y * y - 1] % p,
				ans = ans * sum[0][xr / x * x - 1][yr / y * y - 1] % p;
		res ^= ans;
	}
	io.write(res, '\n');
}
