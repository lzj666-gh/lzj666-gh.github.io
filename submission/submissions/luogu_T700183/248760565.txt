#include<bits/stdc++.h>
#define file(a) freopen(#a".in", "r", stdin), freopen(#a".out", "w", stdout)
//#define int long long
using namespace std;
int n, q; string s;
int32_t main() {
	//file(holiday);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> q >> s;
	if (count(s.begin(), s.end(), '0') == n) {
		while (q--) {
			int k; cin >> k;
			if (k < 2) cout << 0 << endl;
			else cout << k << endl;
		}
		return 0;
	}
	if (count(s.begin(), s.end(), '1') == n) {
		while (q--) {
			if (n < 2) cout << 0 << endl;
			else cout << n << endl;
		}
		return 0;
	}
	if (q == 1) {
		int k, ans = 0; cin >> k;
		for (int i = 0; i < n;) {
			size_t j = s.find_first_not_of(s[i], i);
			if (j > n) j = n;
			if (s[i] == '1' && j - i > 1) ans += j - i;
			i = j;
		}
		if (k == 0) {
			cout << ans << endl;
			return 0;
		}
		if (n == 1) {
			cout << 0 << endl;
			return 0;
		}
		if (n == 2) {
			cout << 2 << endl;
			return 0;
		}
		int mxa = 1;
		for (int i = 0; i < n; ++i) {
			if ((i == 0 || s[i - 1] == '0') && s[i] == '1' && (i == n - 1 || s[i + 1] == '0'))
				mxa = max(mxa, 2);
			if (i > 0 && i < n + 1 && ((i == 1 || s[i - 2] == '0') && s[i - 1] == '1' && s[i] == '0' && s[i + 1] == '1' && (i == n - 2 || s[i + 2] == '0')))
				mxa = max(mxa, 3);
		}
		cout << ans + mxa << endl;
		return 0;
	}
	int l;
	if (n % 4 <= 2) l = n / 4;
	else l = n / 4 + 1;
	while (q--) {
		int k; cin >> k;
		if (k <= l) cout << k * 3 << endl;
		else cout << l * 3 + (k - l) << endl;
	}
	return 0;
}
