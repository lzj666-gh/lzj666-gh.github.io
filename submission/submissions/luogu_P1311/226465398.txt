#include<bits/stdc++.h>
using namespace std;
int n, k, p, a, b;
int sum[200005], cur[200005], frt[200005], ans;
int main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> k >> p;
	for (int i = 1, now = 0; i <= n; ++i) {
		cin >> a >> b;
		if (b <= p) now = i;
		if (now >= frt[a]) sum[a] = cur[a];
		frt[a] = i; ans += sum[a]; cur[a]++;
	}
	cout << ans << endl;
	return 0;
}
