#include<bits/stdc++.h>
using namespace std;
int n, m, a[100000 + 5], bel[100000 + 5];
namespace SOLVE {
    bitset<100000 + 5>ans[40000 + 5], now;
    int cnt[100000 + 5], tot[100000 + 5];
    struct Query {
        int l, r, id;
        friend bool operator<(const Query& a, const Query& b) {
            return bel[a.l] ^ bel[b.l] ? bel[a.l] < bel[b.l] : bel[a.l] & 1 ? a.r < b.r : a.r > b.r;
        }
    }q[120000 + 5];
    void add(int x) {
        cnt[a[x]]++;
        now[a[x] + cnt[a[x]] - 2] = 1;
    }
    void del(int x) {
        now[a[x] + cnt[a[x]] - 2] = 0;
        cnt[a[x]]--;
    }
    void solve(int m) {
        for (int i = 1, j = 1; i <= m; ++i) {
            cin >> q[j].l >> q[j].r;
            tot[i] = q[j].r - q[j].l + 1;
            q[j].id = i; j++;
            cin >> q[j].l >> q[j].r;
            tot[i] += q[j].r - q[j].l + 1;
            q[j].id = i; j++;
            cin >> q[j].l >> q[j].r;
            tot[i] += q[j].r - q[j].l + 1;
            q[j].id = i; j++;
            ans[i].set();
        }
        now.reset();
        for (int i = 1; i <= n; ++i) cnt[i] = 0;
        sort(q + 1, q + m * 3 + 1);
        int l = q[1].l, r = q[1].l - 1;
        for (int i = 1; i <= m * 3; ++i) {
            while (r < q[i].r) add(++r);
            while (l < q[i].l) del(l++);
            while (l > q[i].l) add(--l);
            while (r > q[i].r) del(r--);
            ans[q[i].id] &= now;
        }
        for (int i = 1; i <= m; ++i) {
            cout << tot[i] - ans[i].count() * 3 << endl;
        }
    }
}
int main() {
    cin.tie(0)->sync_with_stdio(false);
    cin >> n >> m; vector<int>tmp;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        tmp.emplace_back(a[i]);
    }
    sort(tmp.begin(), tmp.end());
    // tmp.erase(unique(tmp.begin(), tmp.end()), tmp.end());
    for (int i = 1; i <= n; ++i)
        a[i] = lower_bound(tmp.begin(), tmp.end(), a[i]) - tmp.begin() + 1;
    int t = sqrt(n);
    for (int i = 1; i <= n; ++i)
        bel[i] = i / t + 1;
    int k = 3;
    for (int i = 1; i < k; ++i)
        SOLVE::solve(m / k);
    SOLVE::solve(m - m / k * (k - 1));
    return 0;
}
