struct Ability {
	vector<int>data;
	Ability() :data() {}
}cur;
struct Test {
	Ability need, delta;
	Test() :need(), delta() {}
}; vector<Test>tests;
Ability& operator+=(Ability& now, const Ability& oth) {
	for (int i = 0; i < now.data.size(); ++i) now.data[i] += oth.data[i];
	return now;
}
Ability& operator+=(Ability& now, const Test& oth) { return now += oth.delta; }
bool operator>=(const Ability& a, const Test& b) {
	//cerr << "[debug] 1111 : " << a.data.size() << endl;
	//for (int i = 0; i < a.data.size(); ++i)
	//	cerr << "[debug] : " << b.need.data[i] << endl;
	for (int i = 0; i < a.data.size(); ++i)
		if (a.data[i] < b.need.data[i])
			return /*cerr << "[debug] :" << i << ' ' << a.data[i] << '<' << b.need.data[i] << endl, */0;
	return 1;
}
int n, k;
struct Rand {
	using ull = unsigned long long;
	ull x; Rand() :x(time(0)) {}
	static const ull max() { return -1; };
	static const ull min() { return 0; };
	ull operator()() {
		x ^= x << 5;
		x ^= x >> 11;
		x ^= x << 54;
		return x;
	}
}myrand;
vector<priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>>>q;
bool vis[1000000 + 5];
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> k; cur.data.resize(k); tests.resize(n);
	for (int i = 0; i < n; ++i) tests[i].need.data.resize(k);
	for (int i = 0; i < n; ++i) tests[i].delta.data.resize(k);
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < k; ++j)
			cin >> tests[i].need.data[j];
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < k; ++j)
			cin >> tests[i].delta.data[j];
	q.resize(k);
	for (int i = 0; i < k; ++i)
		for (int j = 0; j < n; ++j)
			q[i].emplace(tests[j].need.data[i], j);
	int ans = 0; queue<int>qq;
	for (int i = 0; i < n; ++i)
		if (ranges::count(tests[i].need.data, 0) == k)
			qq.emplace(i);
	while (!qq.empty()) {
		int now = qq.front(); qq.pop();
		if (vis[now]) continue; vis[now] = 1;
		ans++; cur += tests[now]; // cerr << now << endl;
		for (int i = 0; i < k; ++i) {
			//cerr << "updating " << q[i].top().second << endl;
			if (q[i].empty()) continue;
			int www = q[i].top().second;
			while (cur >= tests[www]) {
				//cerr << "push " << www << endl;
				if (!vis[www])
					qq.emplace(www)/*, vis[www] = 1*/;
				q[i].pop();
				if (q[i].empty()) break;
				www = q[i].top().second;
			}
		}
	}
	cout << ans << endl;
	return 0;
}
