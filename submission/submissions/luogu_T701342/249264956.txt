constexpr int mod = 1e9 + 7;
struct Ability {
	vector<int>data;
	Ability() :data() {}
}cur;
struct Test {
	Ability need, delta;
	Test() :need(), delta() {}
}; vector<Test>tests;
Ability& operator+=(Ability& now, const Ability& oth) {
	for (int i = 0; i < now.data.size(); ++i) now.data[i] += oth.data[i];
	return now;
}
Ability& operator+=(Ability& now, const Test& oth) { return now += oth.delta; }
bool operator>=(const Ability& a, const Test& b) {
	for (int i = 0; i < a.data.size(); ++i)
		if (a.data[i] < b.need.data[i])
			return 0;
	return 1;
}
bool operator<(const Test& a, const Test& b) { return a.need.data == b.need.data ? a.delta.data > b.delta.data:a.need.data < b.need.data; }
int n, k;
struct Rand {
	using ull = unsigned long long;
	ull x; Rand() :x(time(0)) {}
	static const ull max() { return -1; };
	static const ull min() { return 0; };
	ull operator()() {
		x ^= x << 5;
		x ^= x >> 11;
		x ^= x << 54;
		return x;
	}
}myrand;
bool vis[1000000 + 5];
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> k; cur.data.resize(k); tests.resize(n);
	for (int i = 0; i < n; ++i) tests[i].need.data.resize(k);
	for (int i = 0; i < n; ++i) tests[i].delta.data.resize(k);
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < k; ++j)
			cin >> tests[i].need.data[j];
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < k; ++j)
			cin >> tests[i].delta.data[j];
	vector<int>v(k); iota(v.begin(), v.end(), 0);
    mt19937_64 rnd(time(0));
	shuffle(v.begin(), v.end(), rnd);
	for (int i = 0; i < n; ++i) {
		vector<int>vv = tests[i].delta.data;
		for (int j = 0; j < k; ++j)
			tests[i].delta.data[j] = vv[v[j]];
		vv = tests[i].need.data;
		for (int j = 0; j < k; ++j)
			tests[i].need.data[j] = vv[v[j]];
	}
	sort(tests.begin(), tests.end());
	bool flg = 1; int ans = 0;
	while (flg) {
		flg = 0;
		for (int i = 0; i < n; ++i)
			if (cur >= tests[i] && !vis[i])
				cur += tests[i], flg = 1, ans++, vis[i] = 1;
	}
	cout << ans << endl;
	return 0;
}
