#include<bits/stdc++.h>
using namespace std;
int n, k, m, fa[20000 + 1], siz[20000 + 1];
int find(int x) { return fa[x] == x ? x : fa[x] = find(fa[x]); }
bitset<20002>t;
signed main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m >> k;
	for (int i = 1; i <= n; ++i) fa[i] = i, siz[i] = 1;
	for (int i = 1, u, v; i <= k; ++i)
		cin >> u >> v,
		(find(u) != find(v)) && (siz[find(u)] += siz[find(v)], fa[find(v)] = find(u));
	t[0] = 1;
	for (int i = 1; i <= n; ++i)
		if (fa[i] == i)
			t |= t << siz[i];
	int ans = 0;
	for (int i = 1; i <= n; ++i)
		if (t[i] && abs(i - m) < abs(ans - m) || (abs(i - m) == abs(ans - m) && i < ans))
			ans = i;
	cout << ans << endl;
	return 0;
}
