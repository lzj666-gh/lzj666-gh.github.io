#include<bits/stdc++.h>
#define int long long
using namespace std;
int n, T, a[2000000 + 5];
bool check(int t) {
	int sum = 0;
	for (int i = 1; i <= n; ++i) {
		if (a[i] > t) return false;
		int ms = t - a[i];
		sum += min(ms, a[i]) + max(a[i] - ms, 0ll) * 2;
	}
	return sum <= T;
}
int32_t main(int32_t argc, char** argv) {
//	if (argc >= 2) freopen(argv[1], "r", stdin);
//	else freopen("a.in", "r", stdin);
//	if (argc >= 3) freopen(argv[2], "w", stdout);
//	else freopen("a.out", "w", stdout);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> T;
	for (int i = 1; i <= n; ++i) cin >> a[i];
	int l = -1, r = 2 * T + 1, mid;
	if (!check(r)) {
		cout << -1 << endl;
		return 0;
	}
	while (l < r - 1) {
		mid = l + r >> 1;
		if (check(mid)) r = mid;
		else l = mid;
	}
	cout << r << endl;
	return 0;
}
