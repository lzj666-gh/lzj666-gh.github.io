#include<bits/stdc++.h>
using namespace std;
int32_t main() {
    cin.tie(0)->sync_with_stdio(false);
    int n; cin >> n; int ans1 = 0, ans2 = 0;
    for (int i = 1, k; i <= n; ++i) {
        cin >> k;
        if (k == -1) ans1++;
        else if (k == 0) ans2++;
        else {
            cout << ans1 + ans2 << ' ' << ans2 << endl;
            return 0;
        }
    }
    return 0;
}
