#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e7;
int n, m, x, p;
int a[MAXN + 10], bel[MAXN + 10], idx[MAXN + 10];
unsigned int state;
inline unsigned int next_random() {
	state = state ^ (state << 13);
	state = state ^ (state >> 17);
	state = state ^ (state << 5);
	return state;
}
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m >> x >> p >> state;
	for (int i = 1; i <= n; i++) a[i] = next_random() % p;
	vector<vector<int>>pre; pre.emplace_back();
	for (int i = 1; i <= n; ++i) {
		if (pre.back().size() < x) pre.back().push_back(a[i]);
		else pre.emplace_back(), pre.back().push_back(a[i]);
		bel[i] = pre.size() - 1; idx[i] = pre.back().size() - 1;
	}
	vector<vector<int>> bck = pre;
	for (auto& q : pre)
		for (int i = 1; i < q.size(); ++i)
			q[i] = 1ll * q[i - 1] * q[i] % p;
	for (auto& q : bck)
		for (int i = q.size() - 2; i >= 0; --i)
			q[i] = 1ll * q[i + 1] * q[i] % p;
	unsigned long long result = 0;
	int min_len = x, max_len = min(2 * x, n), range_len = max_len - min_len + 1;
	for (int i = 1; i <= m; i++) {
		int len = min_len + (next_random() % range_len);
		int l = 1 + (next_random() % (n - len + 1)), r = l + len - 1;

		int ans = 0;

		if (bel[l] == bel[r]) ans = pre[bel[l]].back();
		else if (bel[l] == bel[r] - 1) ans = 1ll * bck[bel[l]][idx[l]] * pre[bel[r]][idx[r]] % p;
		else if (bel[l] == bel[r] - 2) ans = 1ll * bck[bel[l]][idx[l]] * pre[bel[l] + 1].back() % p * pre[bel[r]][idx[r]] % p;
		else throw logic_error("out of two blocks");

		result ^= 1ULL * ans * i;
	}
	cout << result << endl;
	return 0;
}
