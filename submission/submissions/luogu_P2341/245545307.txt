#include<bits/stdc++.h>
#define int long long
using namespace std;
struct Edge {
	int v, nxt;
}edge[50000 + 5];
int head[10000 + 5], idx;
void add(int u, int v) {
	edge[++idx] = { v,head[u] };
	head[u] = idx;
}
stack<int>stk;
int idk, cnt, scc[10000 + 5], dfn[10000 + 5], low[10000 + 5];
void dfs(int x) {
	low[x] = dfn[x] = ++idk; stk.push(x);
	for (int i = head[x]; i; i = edge[i].nxt)
		if (!dfn[edge[i].v])
			dfs(edge[i].v),
			low[x] = min(low[x], low[edge[i].v]);
		else if (!scc[edge[i].v])
			low[x] = min(low[x], dfn[edge[i].v]);
	if (low[x] == dfn[x]) {
		++cnt; int v;
		while ((v = stk.top()) != x)
			scc[v] = cnt, stk.pop();
		scc[x] = cnt; stk.pop();
	}
}
int n, m;
namespace Final {
	struct Edge {
		int v, nxt;
	}edge[50000 + 5];
	int head[10000 + 5], r[10000 + 5], idx;
	void add(int u, int v) {
		edge[++idx] = { v,head[u] };
		head[u] = idx; r[v]++;
	}
	bitset<10001>b[10001];
	void main() {
		queue<int>q;
		for (int i = 1; i <= cnt; ++i) {
			if (!r[i])
				q.push(i);
			b[i][i] = 1;
		}
		while (!q.empty()) {
			int now = q.front(); q.pop();
			for (int i = head[now]; i; i = edge[i].nxt) {
				b[edge[i].v] |= b[now];
				if (!--r[edge[i].v])
					q.push(edge[i].v);
			}
		}
		int ans = 0;
		for (int i = 1; i <= cnt; ++i)
			ans += b[i].count() == cnt;
		cout << ans << endl;
	}
}
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m;
	for (int i = 1, u, v; i <= m; ++i)
		cin >> u >> v, add(u, v);
	dfs(1);
	//for (int i = 1; i <= n; ++i) cerr << scc[i] << " \n"[i == n];
	for (int i = 1; i <= n; ++i)
		for (int j = head[i]; j; j = edge[j].nxt)
			if (scc[i] ^ scc[edge[j].v])
				Final::add(scc[i], scc[edge[j].v]);
	Final::main();
	return 0;
}
