#include<bits/stdc++.h>
#define int long long
#define file(name) freopen(#name".in", "r", stdin), freopen(#name".out", "w", stdout)
using namespace std;
char mp[128], mp2[128]; string A, B, C;
signed main() {
	//file();
	cin.tie(0)->sync_with_stdio(false);
	cin >> A >> B >> C;
	if (A.size() != B.size()) return cout << "Failed" << endl, 0;
	for (int i = 0; i < A.size(); ++i) {
		if (mp[A[i]] && mp[A[i]] != B[i] || mp2[B[i]] && mp2[B[i]] != A[i] || !isupper(A[i]) || !isupper(B[i])) return cout << "Failed" << endl, 0;
		mp[A[i]] = B[i]; mp2[B[i]] = A[i];
	}
	for (int i = 'A'; i <= 'Z'; ++i) if (!mp[i]) return cout << "Failed" << endl, 0;
	for (const char& c : C) cout << mp[c];
	return 0;
}
