using namespace std;
constexpr int mod = 1e9 + 7;
int n, m, k, pos[1000000 + 5];
int dis[1000000 + 5], v[1000000 + 5];
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m >> k;
	for (int i = 2; i <= n; ++i)
		cin >> dis[i], pos[i] = dis[i] + pos[i - 1];
	for (int i = 1; i <= n; ++i) cin >> v[i], v[i] += v[i - 1];
	vector<int>ans; int anss = 0;
	for (int i = 1; i <= n; ++i) {
		int j = i, curans = 0; vector<int>cur;
		for (int p = 1; p <= m; ++p) {
			if (j > n) break;
			//get first less equal position(deploy position)
			int t = upper_bound(pos + 1, pos + n + 1, pos[j] + k) - pos - 1;
			int q = upper_bound(pos + 1, pos + n + 1, pos[t] + k) - pos - 1;
			curans += v[q] - v[j - 1]; j = q + 1; cur.emplace_back(t);
		}
		if (curans > anss) ans = cur, anss = curans;
		// cerr << "begin pos : " << i << " and get max value " << curans << " and moveto the farthest position : " << j << endl;
	}
	cout << ans.size() << endl;
	for (const int& p : ans) cout << p << ' ';
	return 0;
}
