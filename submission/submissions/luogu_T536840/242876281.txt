#include<bits/stdc++.h>
#include<ext/pb_ds/assoc_container.hpp>
#include<ext/pb_ds/priority_queue.hpp>
#define int long long
#define all(a) (a).begin(), (a).end()
#define file(a) { freopen(#a".in", "r", stdin); freopen(#a".out", "w", stdout); }
using namespace std;
using pi = pair<int, int>;
using pqueue = __gnu_pbds::priority_queue<pi>;
pqueue t;
int n, d, a[1500 + 5], b[1500 + 5];
pqueue::point_iterator it[3000 + 5];
bool del[3000 + 5];
int32_t main() {
	// file(c);
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> d;
	for (int i = 1; i <= n; ++i) {
		cin >> a[i];
		a[i] += a[i - 1];
	}
	for (int i = 1; i <= n; ++i) {
		cin >> b[i];
		b[i] += b[i - 1];
	}
	for (int i = 1; i <= n; ++i) {
		int L = max(1ll, i - d), R = min(n, i + d);
		int sumA = a[R] - a[L - 1];
		int sumB = b[R] - b[L - 1];
		it[i + n] = t.push(make_pair(sumA, i + n));
		it[i] = t.push(make_pair(sumB, i));
	}
	for (int i = n; i >= 1; --i) a[i] -= a[i - 1];
	for (int i = n; i >= 1; --i) b[i] -= b[i - 1];
	int ans = 0;
	while (t.size()) {
		auto k = t.begin();
		if (k == t.end()) break;
		if (k->first <= 0) break;
		int kval = k->first;
		int id = k->second;
		ans += kval;
		if (id <= n) {
			int pos = id;
			int L = max(1ll, pos - d), R = min(n, pos + d);
			for (int j = L; j <= R; ++j) {
				int idx = j + n;
				if (del[idx]) continue;
				pi oldv = *it[idx];
				pi newv = make_pair(oldv.first - b[pos], oldv.second);
				t.modify(it[idx], newv);
			}
		}
		else {
			int pos = id - n;
			int L = max(1ll, pos - d), R = min(n, pos + d);
			for (int j = L; j <= R; ++j) {
				int idx = j;
				if (del[idx]) continue;
				pi oldv = *it[idx];
				pi newv = make_pair(oldv.first - a[pos], oldv.second);
				t.modify(it[idx], newv);
			}
		}
		del[id] = 1;
		t.erase(k);
	}
	cout << ans << endl;
	return 0;
}
