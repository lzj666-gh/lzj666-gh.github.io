#include<bits/stdc++.h>
#define all(a) (a).begin(), (a).end()
#define rall(a) (a).rbegin(), (a).rend()
#define int long long
using namespace std;
int qpow(int a, int b, int mod) {
	int res = 1;
	while (b) {
		if (b & 1) res = res * a % mod;
		a = a * a % mod; b >>= 1;
	}
	return res;
}
int n, a[5000000 + 5];
int fac[10000000 + 5], ifac[10000000 + 5];
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	int n, k, p; cin >> n >> p >> k; fac[0] = 1;
	for (int i = 1; i <= n; ++i) cin >> a[i];
	for (int i = 1; i <= n; ++i) fac[i] = fac[i - 1] * a[i] % p;
	ifac[n] = qpow(fac[n], p - 2, p);
	for (int i = n - 1; i >= 0; --i) ifac[i] = ifac[i + 1] * a[i + 1] % p;
	int ans = 0, d = k;
	for (int i = 1; i <= n; ++i) ans += d * fac[i - 1] % p * ifac[i] % p, d = d * k % p;
	cout << ans % p << endl;
	return 0;
}
