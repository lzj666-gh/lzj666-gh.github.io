#include<bits/stdc++.h>
#define int long long
using namespace std;
vector<int> exkmp(string s) {
	vector<int>res(s.size(), 0);
	for (int i = 1, l = 0, r = 0; i < s.size(); ++i) {
		if (i <= r && res[i - l] < r - i + 1) res[i] = res[i - l];
		else {
			res[i] = max(0, r - i + 1ll);
			while (i + res[i] < s.size() && s[res[i]] == s[i + res[i]]) res[i]++;
		}
		if (i + res[i] - 1 > r) l = i, r = i + res[i] - 1;
	}
	return res;
}
vector<int>exkmp(string a, string b) {
	vector<int>v = move(exkmp(b + ' ' + a));
	return vector<int>(v.begin() + b.size() + 1, v.end());
}
int clac(vector<int> v) {
	int res = 0;
	for (int i = 0; i < v.size(); ++i) {
		res ^= (i + 1) * (v[i] + 1);
	}
	return res;
}
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	string a, b; cin >> a >> b;
	vector<int>res = exkmp(b); res[0] = b.size();
	cout << clac(res) << endl << clac(exkmp(a, b)) << endl;
	return 0;
}
