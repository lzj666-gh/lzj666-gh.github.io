#include<bits/stdc++.h>
using namespace std;
const int MAXN = 33550336 + 5;
int n, m, x, p;
unsigned int state;
inline unsigned int next_random() {
    state = state ^ (state << 13);
    state = state ^ (state >> 17);
    state = state ^ (state << 5);
    return state;
}
int pre_arr[MAXN], bck_arr[MAXN];

int32_t main() {
    cin.tie(0)->sync_with_stdio(false);
    cin >> n >> m >> x >> p >> state;
    int num_blocks = (n + x - 1) / x;
    for (int i = 0; i < n; i++) {
        pre_arr[i] = bck_arr[i] = next_random() % p;
    }
    for (int block = 0; block < num_blocks; block++) {
        int start = block * x;
        int end = min((block + 1) * x, n);
        
        for (int j = start + 1; j < end; j++) {
            pre_arr[j] = 1LL * pre_arr[j - 1] * pre_arr[j] % p;
        }
    }
    for (int block = 0; block < num_blocks; block++) {
        int start = block * x;
        int end = min((block + 1) * x, n);
        
        for (int j = end - 2; j >= start; j--) {
            bck_arr[j] = 1LL * bck_arr[j + 1] * bck_arr[j] % p;
        }
    }
    
    static unsigned int result = 0;
    int min_len = x, max_len = min(2 * x, n), range_len = max_len - min_len + 1;
    
    for (int i = 1; i <= m; i++) {
        int len = min_len + (next_random() % range_len);
        int l = 1 + (next_random() % (n - len + 1));
        int r = l + len - 1;
        
        int ans = 0;
        int bell = (l - 1) / x;
        int belr = (r - 1) / x;
        int idxl = (l - 1) % x;
        int idxr = (r - 1) % x;
        
        if (bell == belr) {
            int block_end = min((bell + 1) * x, n) - 1;
            ans = pre_arr[block_end];
        } else if (bell == belr - 1) {
            ans = 1LL * bck_arr[bell * x + idxl] * pre_arr[belr * x + idxr] % p;
        } else if (bell == belr - 2) {
            int mid_block_end = min((bell + 1) * x, n) - 1;
            ans = 1LL * bck_arr[bell * x + idxl] * pre_arr[mid_block_end] % p * pre_arr[belr * x + idxr] % p;
        } else {
            throw logic_error("out of two blocks");
        }
        
        result ^= ans;
    }
    
    cout << result << endl;
    return 0;
}
