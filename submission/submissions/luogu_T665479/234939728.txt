#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
const int MAXN = 33550336;
int n, m, x, p, N=1;
int a[MAXN + 10],t[MAXN*3];
unsigned int state;
inline unsigned int next_random() {
	state = state ^ (state << 13);
	state = state ^ (state >> 17);
	state = state ^ (state << 5);
	return state;
}
int ans,s,e,l,r,len;
int main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m >> x >> p >> state;
	while(N-2<n)N<<=1;
	for(register int i = 1;i<=N+n;++i)t[i]=1;
	for (register int i = 1; i <= n; ++i){
		a[i] = next_random()%p;
		s=i+N;
		while(s){
			t[s]=1ll*t[s]*a[i]%p;
			s>>=1;			
		}
	}
	
	unsigned int result = 0;
	int range_len = min(x<<1, n) - x + 1;
	for (register int i = 1; i <= m; ++i) {
		len = x + (next_random() % range_len);
		l = 1 + (next_random() % (n - len + 1)), r = l + len - 1;

		ans = 1;
		s=l+N-1,e=r+N+1;
		while(s^e^1){
			if(!(s&1)&&(ans=1ll*ans*t[s^1]%p));
			if(e&1&&(ans=1ll*ans*t[e^1]%p));
			s>>=1,e>>=1;
		}

		result = (result ^ ans );
	}
	cout << result << endl;
	return 0;
}
