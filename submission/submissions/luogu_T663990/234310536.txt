int32_t main() {
//	file(ex_noip2);
	cin.tie(0)->sync_with_stdio(false);
	int T; cin >> T;
	while (T--) {
		string s; cin >> s; int ans = 4;
		for (auto& p : v) {
			if (ans <= p.second) continue;
			for (int i = 0; i + p.first.size() <= s.size(); ++i) {
				bool flg = 1;
				for (int j = 0; j < p.first.size(); ++j) {
					if (p.first[j] == '*') continue;
					flg &= p.first[j] == s[i + j];
				}
				if (flg) ans = min(ans, p.second);
			}
		}
		cout << ans << endl;
	}
	return 0;
}
