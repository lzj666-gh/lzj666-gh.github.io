#include<bits/stdc++.h>
using namespace std;
int n, m; string s;
unordered_map<string, bool>mpp;
unordered_map<string, int>mp;
int main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n;
	for (int i = 1; i <= n; ++i) cin >> s, mpp[s] = 1;
	cin >> m;
	while (m--) {
		cin >> s;
		if (mpp[s])
			if (++mp[s] == 1) cout << "OK" << endl;
			else cout << "REPEAT" << endl;
		else cout << "WRONG" << endl;
	}
	return 0;
}
