using namespace std;
struct Bit {
	bool may[64][2]; int len;
	Bit(unsigned long long x = 0) {
		len = 64;
		for (int i = 0; i < len; ++i)
			may[i][x & 1] = 1, may[i][x & 1 ^ 1] = 0, x >>= 1;
	}
	Bit operator|(const Bit& oth) {
		Bit ans; ans.len = len;
		for (int i = 0; i < len; ++i)
			ans.may[i][1] = may[i][1] || oth.may[i][1],
			ans.may[i][0] = may[i][0] && oth.may[i][0];
		return ans;
	}
	Bit operator&(const Bit& oth) {
		Bit ans; ans.len = len;
		for (int i = 0; i < len; ++i)
			ans.may[i][1] = may[i][1] && oth.may[i][1],
			ans.may[i][0] = may[i][0] || oth.may[i][0];
		return ans;
	}
	friend istream& operator>>(istream& in, Bit& x) {
		string s; in >> s; x.len = s.size();
		for (int i = 0; i < x.len; ++i)
			x.may[i][0] = s[i] == '0' || s[i] == '?',
			x.may[i][1] = s[i] == '1' || s[i] == '?';
		return in;
	}
	friend ostream& operator<<(ostream& out, const Bit& x) {
		out << x.len << ':';
		for (int i = 0; i < x.len; ++i)
			out << (x.may[i][0] ? x.may[i][1] ? '?' : '0' : x.may[i][1] ? '1' : '!');
		return out;
	}
	unsigned long long count() {
		unsigned long long ans = 1;
		for (int i = 0; i < len; ++i)
			ans *= (unsigned long long)may[i][0] + may[i][1];
		return ans;
	}
}a[100000 + 5];
