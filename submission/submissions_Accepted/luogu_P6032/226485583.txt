#include<bits/stdc++.h>
#define int long long
using namespace std;
int n, k, p, a, b;
int sum[2000005], cur[2000005], frt[2000005], ans;
int32_t main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> k >> p;
	for (int i = 1, now = 0; i <= n; ++i) {
		cin >> a >> b;
		if (b <= p) now = i;
		if (now >= frt[a]) sum[a] = cur[a];
		frt[a] = i; ans += sum[a]; cur[a]++;
	}
	cout << ans << endl;
	return 0;
}
