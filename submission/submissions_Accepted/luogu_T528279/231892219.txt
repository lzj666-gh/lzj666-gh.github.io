#include<bits/stdc++.h>
#define file(name) freopen(#name".in", "r", stdin), freopen(#name".out", "w", stdout);
using namespace std;
using ll = long long;
ll n, m, p, q, t[100000 + 5], sum;
int main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m >> p >> q;
	for (int i = 1; i <= n; ++i) cin >> t[i];
	for (ll d = 1, i = 1; i <= n || (cout << d - 1 << endl, 0); d++) {
		ll cur = m, g = round(ceil(1.0 * p * d * m / q));
		if (sum < g) {
			cur -= g - sum;
			sum = g;
		}
		while (cur > t[i] && i <= n) cur -= t[i++];
		sum += cur;
	}
	return 0;
}
