#include<bits/stdc++.h>
using namespace std;
int k, l[6];
signed main() {
	cin.tie(0)->sync_with_stdio(false);
	while (1) {
		cin >> k;
		if (!k) return 0;
		for (int i = 1; i <= k; ++i) cin >> l[i];
		for (int i = k + 1; i <= 5; ++i) l[i] = 0;
		vector<vector<vector<vector<vector<long long>>>>> dp(l[1] + 1,
			vector<vector<vector<vector<long long>>>>(l[2] + 1,
				vector<vector<vector<long long>>>(l[3] + 1,
					vector<vector<long long>>(l[4] + 1,
						vector<long long>(l[5] + 1, 0)))));
		dp[0][0][0][0][0] = 1;
		for (int a = 0; a <= l[1]; ++a)
			for (int b = 0; b <= l[2]; ++b)
				for (int c = 0; c <= l[3]; ++c)
					for (int d = 0; d <= l[4]; ++d)
						for (int e = 0; e <= l[5]; ++e) {
							if (a < l[1]         ) dp[a + 1][b][c][d][e] += dp[a][b][c][d][e];
							if (b < l[2] && a > b) dp[a][b + 1][c][d][e] += dp[a][b][c][d][e];
							if (c < l[3] && b > c) dp[a][b][c + 1][d][e] += dp[a][b][c][d][e];
							if (d < l[4] && c > d) dp[a][b][c][d + 1][e] += dp[a][b][c][d][e];
							if (e < l[5] && d > e) dp[a][b][c][d][e + 1] += dp[a][b][c][d][e];
						}
		cout << dp[l[1]][l[2]][l[3]][l[4]][l[5]] << endl;
	}
	return 0;
}
