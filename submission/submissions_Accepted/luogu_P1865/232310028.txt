using namespace IO;
vector<int>prime;
bool isp[1000000 + 5];
int f[1000000 + 5];
void init() {
	memset(isp, 1, sizeof isp);
	isp[1] = 0;
	for (int i = 2; i <= 1e6; ++i) {
		if (isp[i]) prime.push_back(i);
		for (int& k : prime) {
			if (i * k > 1e6) break;
			isp[i * k] = 0;
			if (i % k == 0) break;
		}
	}
	for (int i = 1; i <= 1e6; ++i) f[i] = f[i - 1] + isp[i];
}
int main() {
	cin.tie(0)->sync_with_stdio(false);
	init();
	int n, m;
	qread(n, m);
	while (n--) {
		int l, r;
		qread(l, r);
		if (!(1 <= l && l <= m) || !(1 <= r && r <= m)) {
			cout << "Crossing the line" << endl;
			continue;
		}
		if (l > r) cout << 0 << endl;
		else cout << f[r] - f[l - 1] << endl;
	}
	return 0;
}
