using namespace std;int a[1000+10],b[1000+10],c[1000+10],sum[1000+10][1000+10],n,C,k;map<int,int> mp;bool check(int mid){for(int x1=1,x2=1;x2<k;x2++){while(c[x2]-c[x1]+1>mid) x1++; for(int y1=1,y2=1;y2<k;y2++){while(c[y2]-c[y1]+1>mid) y1++;if(sum[x2][y2]-sum[x2][y1-1]-sum[x1-1][y2]+sum[x1-1][y1-1]>=C) return true;}}return false;}main(){cin>>C>>n;c[0]=0;k=1;for(int i=0;i<n;i++){cin>>a[i]>>b[i];if(!mp[a[i]]){mp[a[i]]=1;c[k++]=a[i];}if(!mp[b[i]]){mp[b[i]]=1;c[k++]=b[i];}}sort(c,&c[k]);for(int i=0;i<n;i++){int x=lower_bound(c,&c[k],a[i])-c,y=lower_bound(c,&c[k],b[i])-c;sum[x][y]++;}for(int i=1;i<k;i++) for(int j=1;j<k;j++) sum[i][j]+=sum[i-1][j]+sum[i][j-1]-sum[i-1][j-1];int l=1,r=10000;while(l<r){int mid=l+r>>1;if(check(mid)) r=mid;else l=mid+1;}cout<<r<<endl;return 0;}
