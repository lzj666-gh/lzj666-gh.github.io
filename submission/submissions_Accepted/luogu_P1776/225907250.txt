#include<bits/stdc++.h>
using namespace std;
int v[10000000 + 10], w[1000000 + 10];
int dp[10000000 + 10], n, m, cnt;
int main() {
	cin.tie(0)->sync_with_stdio(false);
	cin >> n >> m;
	for (int i = 1; i <= n; ++i) {
		int x, y, s, t = 1;
		cin >> x >> y >> s;
		while (s >= t) {
			v[++cnt] = x * t; w[cnt] = y * t;
			s -= t; t <<= 1;
		}
		if (s) v[++cnt] = x * s, w[cnt] = y * s;
	}
	for (int i = 1; i <= cnt; ++i)
		for (int j = m; j >= w[i]; --j)
			dp[j] = max(dp[j], dp[j - w[i]] + v[i]);
	cout << dp[m] << endl;
	return 0;
}
